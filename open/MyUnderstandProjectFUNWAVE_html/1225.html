46     !    WAVE_BREAKING is subroutine to do wave breaking 
47     !      using the artificial eddy viscosity scheme 
48     !      For the shock-capturing based breaking scheme 
49     !      this subroutine is only used for
50     !      demonstration or calculating bubbles or foam as 
51     !      SHOW_BREAKING = T.
52     !    
53     !    HISTORY: 
54     !      11/22/2010 Fengyan Shi
55     !      04/15/2015 Fengyan Shi, added viscosity breaking
56     !      09/19/2015 YoungKwang Chio, added viscosity distribution within WaveMaker region
57     !
58     !-------------------------------------------------------------------------------------
59     
60     SUBROUTINE WAVE_BREAKING
61          USE GLOBAL
62          IMPLICIT NONE
63     
64           IF(SHOW_BREAKING)THEN
65             CALL BREAKING(Mloc,Nloc,ETAx,ETAy,ETAT,Cbrk1,Cbrk2,H,MinDepthFrc,DT,&
66                    DX,DY,T_brk,AGE_BREAKING)
67     
68     	ELSEIF( WAVEMAKER_VIS )THEN
69     	  ! ykchoi(08.19.2015)
70     	  !
71             CALL VISCOSITY_WMAKER(Mloc,Nloc,ETAT,visbrk,H,MinDepthFrc,DX,DY)
72           ENDIF
73     
74     
75     END SUBROUTINE WAVE_BREAKING
76     
77     !-------------------------------------------------------------------------------------
78     !
79     !    BREAKING is subroutine to calculate viscosity and breaking age
80     !    for the artificial eddy viscosity breaking scheme
81     !    
82     !    HISTORY: 
83     !     11/22/2010 Fengyan Shi
84     !
85     !-------------------------------------------------------------------------------------
86     
87     SUBROUTINE BREAKING(M,N,ETAx,ETAy,ETAT,Cbrk1,Cbrk2,H,MinDepthFrc,&
88                    DT,DX,DY,T_brk,AGE)
89          USE PARAM
90          USE GLOBAL,ONLY : Depth,nu_break,ETA,Nghost,Xc_WK,Yc_WK,Ibeg,Iend,&
91                            Jbeg,Jend,Width_WK,Xc_WK,Yc_WK,Ywidth_WK,WAVEMAKER_Cbrk,nu_bkg
92     # if defined(PARALLEL)
93          USE GLOBAL,ONLY : npx,npy
94     # endif
95          IMPLICIT NONE
96          INTEGER,INTENT(IN)::M,N
97          REAL(SP),INTENT(IN)::Cbrk1,Cbrk2,MinDepthFrc,DT,T_brk
98          REAL(SP) :: cap1,cap2
99          REAL(SP) :: xmk,ymk,DXg,DYg
100     !     REAL(SP) :: lim_breaker=0.10
101     # if defined (CARTESIAN)
102          REAL(SP),INTENT(IN)::DX,DY
103     # else
104          REAL(SP),DIMENSION(M,N),INTENT(IN)::DX,DY
105     # endif
106          REAL(SP),DIMENSION(M,N),INTENT(IN)::ETAx,ETAy,ETAt,H
107          REAL(SP),DIMENSION(M,N),INTENT(OUT)::AGE
108          REAL(SP)::C,Angle,AGE1,AGE2,AGE3,propx,propy,propxy
109     
110          DO J=Nghost+1,N-Nghost
111          DO I=Nghost+1,M-Nghost
112     
113          tmp3=SQRT(GRAV*MAX(MinDepthFrc,H(I,J)))
114          tmp1=Cbrk1*tmp3
115          tmp2=Cbrk2*tmp3
116          IF(ETAt(I,J).GE.tmp1.AND.(  &
117            AGE(I,J).EQ.ZERO.OR.AGE(I,J).GT.T_brk))THEN
118           AGE(I,J)=DT
119          ELSE
120           IF(AGE(I,J).GT.ZERO)THEN
121             AGE(I,J)=AGE(I,J)+DT
122           ELSE
123             tmp1=MAX(SQRT(ETAx(I,J)*ETAx(I,J)+ETAy(I,J)*ETAy(I,J)),SMALL)
124             C=MIN(ABS(ETAt(I,J))/tmp1,SQRT(GRAV*ABS(H(I,J))))
125     ! propagation time between a dx, dy and ds
126     # if defined (CARTESIAN)
127             DXg=DX
128             DYg=DY
129             propxy=SQRT(DX*DX+DY*DY)/MAX(C,SMALL)
130             propx=SQRT(DX*DX)/MAX(C,SMALL)
131             propy=SQRT(DY*DY)/MAX(C,SMALL)
132     # else
133             DXg=DX(I,J)
134             DYg=DY(I,J)
135             propxy=SQRT(DX(I,J)*DX(I,J)+DY(I,J)*DY(I,J))/MAX(C,SMALL)
136             propx=SQRT(DX(I,J)*DX(I,J))/MAX(C,SMALL)
137             propy=SQRT(DY(I,J)*DY(I,J))/MAX(C,SMALL)
138     # endif
139             ANGLE=ATAN2(ETAy(I,J),ETAx(I,J))
140     
141             IF(ETAt(I,J).GE.tmp2)THEN
142     ! 4 quadrants 
143     ! quadrant 1
144              IF(ANGLE.GE.ZERO.AND.ANGLE.LT.90.0_SP)THEN
145                AGE1=AGE(I-1,J)
146                AGE2=AGE(I-1,J-1)
147                AGE3=AGE(I,J-1)
148                IF((AGE1>=DT.AND.AGE1>propx).OR.&
149                   (AGE2>=DT.AND.AGE2>propxy).OR.&
150                   (AGE3>=DT.AND.AGE3>propy))THEN
151                 AGE(I,J)=DT
152                ENDIF         
153              ENDIF
154     ! quadrant 2
155              IF(ANGLE.GE.90.0_SP.AND.ANGLE.LT.180.0_SP)THEN
156                AGE1=AGE(I+1,J)
157                AGE2=AGE(I+1,J-1)
158                AGE3=AGE(I,J-1)
159                IF((AGE1>=DT.AND.AGE1>propx).OR.&
160                   (AGE2>=DT.AND.AGE2>propxy).OR.&
161                   (AGE3>=DT.AND.AGE3>propy))THEN
162                 AGE(I,J)=DT
163                ENDIF         
164              ENDIF
165     ! quadrant 3
166              IF(ANGLE.GE.-180.0_SP.AND.ANGLE.LT.-90.0_SP)THEN
167                AGE1=AGE(I+1,J)
168                AGE2=AGE(I+1,J+1)
169                AGE3=AGE(I,J+1)
170                IF((AGE1>=DT.AND.AGE1>propx).OR.&
171                   (AGE2>=DT.AND.AGE2>propxy).OR.&
172                   (AGE3>=DT.AND.AGE3>propy))THEN
173                 AGE(I,J)=DT
174                ENDIF         
175              ENDIF
176     ! quadrant 4
177              IF(ANGLE.GE.-90.0_SP.AND.ANGLE.LT.0.0_SP)THEN
178                AGE1=AGE(I,J+1)
179                AGE2=AGE(I-1,J+1)
180                AGE3=AGE(I-1,J)
181                IF((AGE1>=DT.AND.AGE1>propy).OR.&
182                   (AGE2>=DT.AND.AGE2>propxy).OR.&
183                   (AGE3>=DT.AND.AGE3>propx))THEN
184                 AGE(I,J)=DT
185                ENDIF         
186              ENDIF
187     
188            ENDIF
189     
190           ENDIF
191          ENDIF 
192     
193     ! set viscosity
194     
195     ! wavemaker
196     # if defined (PARALLEL)
197                 xmk=(I-Ibeg)*DXg+npx*(M-2*Nghost)*DXg
198                 ymk=(J-Jbeg)*DYg+npy*(N-2*Nghost)*DYg
199     # else
200                 xmk=(I-Ibeg)*DXg
201                 ymk=(J-Jbeg)*DYg
202     # endif
203     
204     ! wavemaker doesnt use breaker age
205     
206         IF(ABS(xmk-Xc_WK)<Width_WK.AND. &
207                 ABS(ymk-Yc_WK)<Ywidth_WK/2.0_SP)THEN
208     
209           IF(ETAt(I,J)>MIN(tmp2,WAVEMAKER_Cbrk*tmp3))THEN
210              cap1=1.0*(MAX(Depth(I,J),MinDepthFrc)+ETA(I,J))
211              nu_break(I,J)=cap1*WAVEMAKER_Cbrk*tmp3+nu_bkg
212           ELSE
213              nu_break(I,J)=ZERO+nu_bkg
214           ENDIF
215     
216         ELSE ! outside wavemaker    
217     
218          IF(AGE(I,J)>ZERO.AND.AGE(I,J)<T_brk.AND.ETAt(I,J)>tmp2)THEN
219            cap1=1.0*(MAX(Depth(I,J),MinDepthFrc)+ETA(I,J))
220     !   note 09/17/2016
221     !   Kennedy et al used a transition for nu, the transition is basically
222     !   used to avoid instabilities. Choi and I did some tests:
223     !   1) use a static nu instead of a function of eta_t which 
224     !      causes a numerical instability. However, instabilities still
225     !      occur for some 2D cases. 
226     !   2) use a static nu as a function of Cbrk2. 
227     !      nu_break(I,J) = cap1*tmp2 + nu_bkg
228     !      This is the setting
229     !      I used before the transition scheme. It worked well but
230     !      we should re-calibrate Cbrk2. 
231     !   
232     # if defined (VIS_KENNEDY)
233             nu_break(I,J)=cap1*ABS(ETAt(I,J))+nu_bkg
234     # elif defined (VIS_STATIC_TRANS)
235              nu_break(I,J)=cap1*tmp3*(Cbrk2+(Cbrk1-Cbrk2)*(T_brk-AGE(I,J))/T_brk)+nu_bkg
236     # else
237             nu_break(I,J) = cap1*tmp2 + nu_bkg
238     # endif
239          ELSE
240            nu_break(I,J)=ZERO+nu_bkg
241          ENDIF
242               
243        ENDIF ! end wavemaker
244     
245          ENDDO
246          ENDDO
247     
248     END SUBROUTINE BREAKING
