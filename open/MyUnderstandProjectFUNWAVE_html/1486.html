46     !    ESTIMATE_HUV is subroutine to calculate eta, ubar and vbar
47     !      using 3rd-order LK scheme 
48     !
49     !    HISTORY: 
50     !      05/12/2011 Fengyan Shi
51     !
52     !-------------------------------------------------------------------------------------
53     SUBROUTINE ESTIMATE_HUV(ISTEP)
54          USE GLOBAL
55          IMPLICIT NONE
56          INTEGER,INTENT(IN)::ISTEP
57          REAL(SP),PARAMETER::n_left=-1.0_SP,n_right=1.0_SP,n_bottom=-1.0_SP,n_top=1.0_SP
58          REAL(SP)::F_left,F_right,F_bottom,F_top,WK_Source
59          REAL(SP),DIMENSION(Ibeg:Iend,Jbeg:Jend)::R1,R2,R3
60     ! now work for spherical # if defined (CARTESIAN)
61          REAL(SP)::xmk,ymk
62     ! now work for spherical # endif
63          REAL(SP)::DXg,DYg
64     
65          INTEGER::kf,kd
66     
67     ! MUSCL-Hancock, Zhou et al., p. 7
68     
69     # if defined (CARTESIAN)
70          DXg=DX
71          DYg=DY
72     # else
73     ! only for wavemaker
74          DXg=DX(1,1)
75          DYg=DY(1,1)
76     # endif
77     
78     ! solve eta
79          DO J=Jbeg,Jend
80          DO I=Ibeg,Iend
81           F_left=P(I,J)
82           F_right=P(I+1,J)
83           F_bottom=Q(I,J)
84           F_top=Q(I,J+1)
85     ! now work for spherical # if defined (CARTESIAN)
86           IF(WAVEMAKER(1:6)=='WK_IRR')THEN
87     # if defined (PARALLEL)
88                 xmk=(I-Ibeg)*DXg+npx*(Mloc-2*Nghost)*DXg
89                 ymk=(J-Jbeg)*DYg+npy*(Nloc-2*Nghost)*DYg
90     # else
91                 xmk=(I-Ibeg)*DXg
92                 ymk=(J-Jbeg)*DYg
93     # endif
94              IF(ABS(xmk-Xc_WK)<Width_WK.AND. &
95                 ABS(ymk-Yc_WK)<Ywidth_WK/2.0_SP)THEN
96               WK_Source=ZERO
97               DO kf=1,Nfreq
98                WK_Source=WK_Source+TANH(PI/(Time_ramp/FreqPeak)*TIME)*(Cm(I,J,kf) &
99                            *COS(OMGN_IR(KF)*TIME) &
100                            +Sm(I,J,kf)*SIN(OMGN_IR(KF)*TIME))
101               ENDDO
102     
103               R1(I,J)=-1.0_SP/DXg*(F_right*n_right+F_left*n_left) &
104                     -1.0_SP/DYg*(F_top*n_top+F_bottom*n_bottom) &
105             ! wavemaker
106                     +WK_Source      
107              ELSE
108              R1(I,J)=-1.0_SP/DXg*(F_right*n_right+F_left*n_left) &
109                        -1.0_SP/DYg*(F_top*n_top+F_bottom*n_bottom)
110              ENDIF
111            ELSEIF(WAVEMAKER(1:6)=='WK_REG')THEN
112     # if defined (PARALLEL)
113                 xmk=(I-Ibeg)*DXg+npx*(Mloc-2*Nghost)*DXg
114                 ymk=(J-Jbeg)*DYg+npy*(Nloc-2*Nghost)*DYg
115     # else
116                 xmk=(I-Ibeg)*DXg
117                 ymk=(J-Jbeg)*DYg
118     # endif
119              IF(ABS(xmk-Xc_WK)<Width_WK.AND. &
120                 ABS(ymk-Yc_WK)<Ywidth_WK/2.0_SP)THEN
121               
122               R1(I,J)=-1.0_SP/DXg*(F_right*n_right+F_left*n_left) &
123                     -1.0_SP/DYg*(F_top*n_top+F_bottom*n_bottom) &
124             ! wavemaker 
125                      +WaveMaker_Mass(I,J)    
126              ELSE
127              R1(I,J)=-1.0_SP/DXg*(F_right*n_right+F_left*n_left) &
128                        -1.0_SP/DYg*(F_top*n_top+F_bottom*n_bottom)
129              ENDIF
130            ELSEIF(WAVEMAKER(1:7)=='WK_TIME')THEN
131     # if defined (PARALLEL)
132                 xmk=(I-Ibeg)*DXg+npx*(Mloc-2*Nghost)*DXg
133                 ymk=(J-Jbeg)*DYg+npy*(Nloc-2*Nghost)*DYg
134     # else
135                 xmk=(I-Ibeg)*DXg
136                 ymk=(J-Jbeg)*DYg
137     # endif
138              IF(ABS(xmk-Xc_WK)<Width_WK.AND. &
139                 ABS(ymk-Yc_WK)<Ywidth_WK/2.0_SP)THEN
140               
141               R1(I,J)=-1.0_SP/DXg*(F_right*n_right+F_left*n_left) &
142                     -1.0_SP/DYg*(F_top*n_top+F_bottom*n_bottom) &
143                     +WaveMaker_Mass(I,J)      
144              ELSE
145              R1(I,J)=-1.0_SP/DXg*(F_right*n_right+F_left*n_left) &
146                        -1.0_SP/DYg*(F_top*n_top+F_bottom*n_bottom)
147              ENDIF   
148     
149            ELSEIF(WAVEMAKER(1:9)=='WK_DATA2D')THEN
150     # if defined (PARALLEL)
151                 xmk=(I-Ibeg)*DXg+npx*(Mloc-2*Nghost)*DXg
152                 ymk=(J-Jbeg)*DYg+npy*(Nloc-2*Nghost)*DYg
153     # else
154                 xmk=(I-Ibeg)*DXg
155                 ymk=(J-Jbeg)*DYg
156     # endif
157              IF(ABS(xmk-Xc_WK)<Width_WK.AND. &
158                 ABS(ymk-Yc_WK)<Ywidth_WK/2.0_SP)THEN!
159               
160               R1(I,J)=-1.0_SP/DXg*(F_right*n_right+F_left*n_left) &
161                     -1.0_SP/DYg*(F_top*n_top+F_bottom*n_bottom) &
162                     +WaveMaker_Mass(I,J)      
163              ELSE
164              R1(I,J)=-1.0_SP/DXg*(F_right*n_right+F_left*n_left) &
165                        -1.0_SP/DYg*(F_top*n_top+F_bottom*n_bottom)
166              ENDIF 
167        
168           ELSE ! no wk_wavemaker, theres bug in version 1.1 Dxg,Dyg should be 
169                ! replaced by Dxg() and Dy()
170     # if defined (CARTESIAN)
171             R1(I,J)=-1.0_SP/DXg*(F_right*n_right+F_left*n_left) &
172                        -1.0_SP/DYg*(F_top*n_top+F_bottom*n_bottom)
173     # else
174             R1(I,J)=-1.0_SP/DX(I,J)*(F_right*n_right+F_left*n_left) &
175                        -1.0_SP/DY(I,J)*(F_top*n_top+F_bottom*n_bottom)
176     # endif
177           ENDIF
178     # if defined (CARTESIAN)
179     ! do nothing
180     # else
181     
182     # if defined (ZALPHA)
183     ! extra terms for z_alpha: 1/R tan theta (HV+hV4)
184             R1(I,J)=R1(I,J)+1.0_SP/R_earth*TAN(Lat_theta(I,J))*(HV(I,J)+ &
185                     Depth(I,J)*V4(I,J)*Gamma1)
186     # else
187     ! extra terms for depth-averaged u: 1/R tan theta HV
188             R1(I,J)=R1(I,J)+1.0_SP/R_earth*TAN(Lat_theta(I,J))*HV(I,J)
189     # endif
190     
191     # endif
192     
193     # if defined (ITERATION)
194           Eta(I,J)=ALPHA(ISTEP)*Eta0(I,J)+BETA(ISTEP)*(EtaOld(I,J)+DT*R1(I,J))
195     # else
196           Eta(I,J)=ALPHA(ISTEP)*Eta0(I,J)+BETA(ISTEP)*(Eta(I,J)+DT*R1(I,J))
197     
198     ! eta_limiter is used for the case as wave touches the seabed within wavemaker
199           IF(ETA_LIMITER)THEN
200             IF(Eta(I,J)<TroughLimit)Eta(I,J)=TroughLimit
201             IF(Eta(I,J)>CrestLimit)Eta(I,J)=CrestLimit
202           ENDIF ! end eta_limiter
203     
204     # endif
205          ENDDO
206          ENDDO
207     
208     ! solve ubar
209          DO J=Jbeg,Jend
210          DO I=Ibeg,Iend
211           F_left=Fx(I,J)
212           F_right=Fx(I+1,J)
213           F_bottom=Fy(I,J)
214           F_top=Fy(I,J+1)
215     # if defined (CARTESIAN)
216           R2(I,J)=-1.0_SP/DX*(F_right*n_right+F_left*n_left) &
217                            -1.0_SP/DY*(F_top*n_top+F_bottom*n_bottom) &
218                             +SourceX(I,J)
219     # else
220           R2(I,J)=-1.0_SP/DX(I,J)*(F_right*n_right+F_left*n_left) &
221                            -1.0_SP/DY(I,J)*(F_top*n_top+F_bottom*n_bottom) &
222                             +SourceX(I,J)
223     # endif
224     # if defined (ITERATION)
225           Ubar(I,J)=ALPHA(ISTEP)*Ubar0(I,J)+BETA(ISTEP)*(UbarOld(I,J)+DT*R2(I,J))
226     # else
227           Ubar(I,J)=ALPHA(ISTEP)*Ubar0(I,J)+BETA(ISTEP)*(Ubar(I,J)+DT*R2(I,J))
228     # endif
229          ENDDO
230          ENDDO
231     
232     ! solve vbar
233          DO J=Jbeg,Jend
234          DO I=Ibeg,Iend
235           F_left=Gx(I,J)
236           F_right=Gx(I+1,J)
237           F_bottom=Gy(I,J)
238           F_top=Gy(I,J+1)
239     # if defined (CARTESIAN)
240           R3(I,J)=-1.0_SP/DX*(F_right*n_right+F_left*n_left) &
241                            -1.0_SP/DY*(F_top*n_top+F_bottom*n_bottom) &
242                            +SourceY(I,J)
243     # else
244           R3(I,J)=-1.0_SP/DX(I,J)*(F_right*n_right+F_left*n_left) &
245                            -1.0_SP/DY(I,J)*(F_top*n_top+F_bottom*n_bottom) &
246                            +SourceY(I,J)
247     # endif
248     # if defined (ITERATION)
249           Vbar(I,J)=ALPHA(ISTEP)*Vbar0(I,J)+BETA(ISTEP)*(VbarOld(I,J)+DT*R3(I,J))
250     # else
251           Vbar(I,J)=ALPHA(ISTEP)*Vbar0(I,J)+BETA(ISTEP)*(Vbar(I,J)+DT*R3(I,J))
252     # endif
253          ENDDO
254          ENDDO
255     
256          CALL GET_Eta_U_V_HU_HV
257     
258     END SUBROUTINE ESTIMATE_HUV
259     
260     !-------------------------------------------------------------------------------------
261     !
262     !    GET_Eta_U_V_HU_HV is subroutine to obtain Eta, u,v,hu,hv
263     !
264     !  HISTORY: 
265     !       09/17/2010 Fengyan Shi
266     !       10/14/2012 Fengyan Shi, added nesting bc
267     !       08/06/2015 Young-Kwang Choi, modified U0 V0 shift 
268     !       01/27/2016 Fengyan Shi, made serial/parallel codes consistent
269     !                               added parallel code of periodic bc in y
270     !
271     !-------------------------------------------------------------------------------------
272     SUBROUTINE GET_Eta_U_V_HU_HV
273          USE GLOBAL
274          IMPLICIT NONE
275          REAL(SP)::Fr,Utotal,Utheta,dep,depl,depr,reta,retal,retar
276          REAL(SP),DIMENSION(Mloc,Nloc) :: myA,myC,myD,myF
277     # if defined(PARALLEL)
278          REAL(SP),DIMENSION(Nglob) :: AperG,BperG,CperG,DperG,VperG
279          REAL(SP),DIMENSION(Mglob,Nglob) ::glbA,glbC,glbD
280          REAL(SP),DIMENSION(Mglob+2*Nghost,Nglob+2*Nghost)::VG
281     # else
282          REAL(SP),DIMENSION(Mloc) :: Axx,Cxx,Dxx
283          REAL(SP),DIMENSION(Mloc) :: U1D
284          REAL(SP),DIMENSION(Nloc) :: Ayy,Cyy,Dyy
285          REAL(SP),DIMENSION(Nloc) :: V1D
286     # if defined (CARTESIAN)
287          REAL(SP),DIMENSION(Jend-Jbeg+1) :: Aper,Bper,Cper,Dper,Vper 
288     # endif 
289     # endif 
290           INTEGER :: IM
291     
292     ! calculate etar, u and vetar, HU, HV
293          H=Eta*Gamma3+Depth
294     
295     
296     !   tridiagonal coefficient
297     ! x direction
298     
299     ! shift U and V
300     !     U0=U    !ykchoi (15. 08. 06.) 
301     !     V0=V    !ykchoi
302     ! ykchoi : U0, V0 are moved to the above part of "Do istage=1,3"
303     
304        IF(DISPERSION)THEN
305     
306          DO J=Jbeg,Jend
307          DO I=Ibeg,Iend
308            dep=Max(Depth(I,J),MinDepthFrc)
309            depl=Max(Depth(I-1,J),MinDepthFrc)
310            depr=Max(Depth(I+1,J),MinDepthFrc)
311     
312     # if defined (CARTESIAN)
313            tmp1=Gamma1*MASK9(I,J)*(b1/2.0_SP/DX/DX*dep*dep + b2/DX/DX*depl*dep)
314            tmp2=1.0_SP+Gamma1*MASK9(I,J)*(-b1/DX/DX*dep*dep-2.0_SP*b2/DX/DX*dep*dep)
315            tmp3=Gamma1*MASK9(I,J)*(b1/2.0_SP/DX/DX*dep*dep + b2/DX/DX*dep*depr)
316            tmp4=Ubar(I,J)*MASK(I,J)/Max(H(I,J),MinDepthFrc)  &
317                 + Gamma1*MASK9(I,J)*( -b1/2.0_SP*dep*dep*Vxy(I,J)-b2*dep*DVxy(I,J))
318     # else
319            tmp1=Gamma1*MASK9(I,J)*(b1/2.0_SP/DX(I,J)/DX(I,J)*dep*dep + b2/DX(I,J)/DX(I,J)*depl*dep)
320            tmp2=1.0_SP+Gamma1*MASK9(I,J)*(-b1/DX(I,J)/DX(I,J)*dep*dep-2.0_SP*b2/DX(I,J)/DX(I,J)*dep*dep)
321            tmp3=Gamma1*MASK9(I,J)*(b1/2.0_SP/DX(I,J)/DX(I,J)*dep*dep + b2/DX(I,J)/DX(I,J)*dep*depr)
322            tmp4=Ubar(I,J)*MASK(I,J)/Max(H(I,J),MinDepthFrc)  &
323                 + Gamma1*MASK9(I,J)*( -b1/2.0_SP*dep*dep*Vxy(I,J)-b2*dep*DVxy(I,J)) 
324     ! remember to document this part for spherical version 
325     # endif
326     
327     ! I added coupling condition 10/14/2012
328     
329     # if defined (COUPLING)
330          IF(IN_DOMAIN_WEST)THEN
331            IF(I.eq.Ibeg)THEN
332              tmp4=tmp4-tmp1*U(I-1,J)
333            ENDIF
334          ELSEIF(IN_DOMAIN_EAST)THEN
335            IF(I.eq.Iend)THEN
336              tmp4=tmp4-tmp3*U(I+1,J)
337            ENDIF
338          ENDIF
339     # endif
340     
341     
342            IF(tmp2.NE.0.0_SP.OR.MASK(I,J).GT.0)THEN
343               myA(I,J)=tmp1/tmp2
344               myC(I,J)=tmp3/tmp2
345               myD(I,J)=tmp4/tmp2
346            ELSE
347               myA(I,J)=ZERO
348               myC(I,J)=ZERO
349               myD(I,J)=ZERO
350            ENDIF
351          ENDDO
352          ENDDO
353     
354     # if defined (PARALLEL)
355          call TRIDx(myA,myC,myD,myF)
356          U(Ibeg:Iend,Jbeg:Jend) = myF(Ibeg:Iend,Jbeg:Jend)
357     # else 
358          call TRIDx_ser(myA,myC,myD,myF)
359          U(Ibeg:Iend,Jbeg:Jend) = myF(Ibeg:Iend,Jbeg:Jend)
360     # endif
361     
362     ! y direction
363     # if defined (PARALLEL)
364          myA=ZERO
365          myC=ZERO
366          myD=ZERO
367     # endif
368          DO I=Ibeg,Iend
369          DO J=Jbeg,Jend
370            dep=Max(Depth(I,J),MinDepthFrc)
371            depl=Max(Depth(I,J-1),MinDepthFrc)
372            depr=Max(Depth(I,J+1),MinDepthFrc)
373     
374     # if defined (CARTESIAN)
375          IF(DISP_TIME_LEFT)THEN
376            reta=Eta(I,J)
377            retal=Eta(I,J-1)
378            retar=Eta(I,J+1)
379            tmp1=Gamma1*MASK9(I,J)*(b1/2.0_SP/DY/DY*dep*dep + b2/DY/DY*depl*dep) &
380                  -Gamma2*MASK9(I,J)*((reta+retal)*depl/2.0_SP/DY/DY+(retal+reta)*(retal+reta)/8.0_SP/DY/DY)
381            tmp2=1.0_SP+Gamma1*MASK9(I,J)*(-b1/DY/DY*dep*dep-2.0_SP*b2/DY/DY*dep*dep) &
382                  +Gamma2*MASK9(I,J)*((retar+retal+2.0_SP*reta)/2.0_SP/DY/DY &
383                             +(retar*retar+2.0_SP*reta*reta &
384                               +2.0_SP*reta*retar+2.0_SP*retal*reta+retal*retal)/8.0_SP/DY/DY)
385            tmp3=Gamma1*MASK9(I,J)*(b1/2.0_SP/DY/DY*dep*dep + b2/DY/DY*dep*depr) &
386                  -Gamma2*MASK9(I,J)*((reta+retar)*depr/2.0_SP/DY/DY+(retar+reta)*(retar+reta)/8.0_SP/DY/DY)
387            tmp4=Vbar(I,J)*MASK(I,J)/Max(H(I,J),MinDepthFrc)  &
388                  + Gamma1*MASK9(I,J)*(-b1/2.0_SP*dep*dep*Uxy(I,J)-b2*dep*DUxy(I,J)) &
389                 + Gamma2*MASK9(I,J)*(reta*reta/2.0_SP*Uxy(I,J)+reta*DUxy(I,J) &
390                                  + reta*ETAy(I,J)*Ux(I,J) + ETAy(I,J)*DUx(I,J) )
391          ELSE
392            tmp1=Gamma1*MASK9(I,J)*(b1/2.0_SP/DY/DY*dep*dep + b2/DY/DY*depl*dep) 
393            tmp2=1.0_SP+Gamma1*MASK9(I,J)*(-b1/DY/DY*dep*dep-2.0_SP*b2/DY/DY*dep*dep) 
394            tmp3=Gamma1*MASK9(I,J)*(b1/2.0_SP/DY/DY*dep*dep + b2/DY/DY*dep*depr)
395            tmp4=Vbar(I,J)*MASK(I,J)/Max(H(I,J),MinDepthFrc)  &
396                  + Gamma1*MASK9(I,J)*(-b1/2.0_SP*dep*dep*Uxy(I,J)-b2*dep*DUxy(I,J))
397          ENDIF  
398     # else
399            tmp1=Gamma1*MASK9(I,J)*(b1/2.0_SP/DY(I,J)/DY(I,J)*dep*dep + b2/DY(I,J)/DY(I,J)*depl*dep) 
400            tmp2=1.0_SP+Gamma1*MASK9(I,J)*(-b1/DY(I,J)/DY(I,J)*dep*dep-2.0_SP*b2/DY(I,J)/DY(I,J)*dep*dep) 
401            tmp3=Gamma1*MASK9(I,J)*(b1/2.0_SP/DY(I,J)/DY(I,J)*dep*dep + b2/DY(I,J)/DY(I,J)*dep*depr)
402            tmp4=Vbar(I,J)*MASK(I,J)/Max(H(I,J),MinDepthFrc)  &
403                  + Gamma1*MASK9(I,J)*(-b1/2.0_SP*dep*dep*Uxy(I,J)-b2*dep*DUxy(I,J)) 
404     ! remember to document this part for spherical version
405     # endif
406     
407     # if defined (COUPLING)
408          IF(IN_DOMAIN_SOUTH)THEN
409            IF(J.eq.Jbeg)THEN
410              tmp4=tmp4-tmp1*V(I,J-1)
411            ENDIF
412          ELSEIF(IN_DOMAIN_SOUTH)THEN
413            IF(J.eq.Jend)THEN
414              tmp4=tmp4-tmp3*V(I,J+1)
415            ENDIF
416          ENDIF
417     # endif
418       
419            IF(tmp2.NE.0.0_SP.OR.MASK(I,J).GT.0)THEN
420              myA(I,J)=tmp1/tmp2
421              myC(I,J)=tmp3/tmp2
422              myD(I,J)=tmp4/tmp2
423            ELSE
424              myA(I,J)=ZERO
425              myC(I,J)=ZERO
426              myD(I,J)=ZERO
427            ENDIF
428          ENDDO
429          ENDDO ! end I
430     
431          IF(PERIODIC) THEN
432     
433     !  Sherman-Morrison algorithm 01/27/2016 fyshi
434     
435           CALL TRIDy_periodic (myA,myC,myD,myF)  
436            V(Ibeg:Iend,Jbeg:Jend) = myF(Ibeg:Iend,Jbeg:Jend)
437          ELSE ! no periodic
438     # if defined (PARALLEL)
439            CALL TRIDy(myA,myC,myD,myF)
440            V(Ibeg:Iend,Jbeg:Jend) = myF(Ibeg:Iend,Jbeg:Jend)
441     # else
442            CALL TRIDy_ser(myA,myC,myD,myF)
443            V(Ibeg:Iend,Jbeg:Jend) = myF(Ibeg:Iend,Jbeg:Jend)
444     # endif
445     
446          ENDIF ! end if periodic
447     
448        ELSE  ! if no dispersion
449          DO J=Jbeg,Jend
450          DO I=Ibeg,Iend  
451             U(I,J)=Ubar(I,J)/Max(H(I,J),MinDepthFrc)
452             V(I,J)=Vbar(I,J)/Max(H(I,J),MinDepthFrc)
453          ENDDO
454          ENDDO   
455     
456        ENDIF  ! end dispersion
457     
458          DO J=Jbeg,Jend
459          DO I=Ibeg,Iend   
460            IF(MASK(I,J)<1)THEN
461             Ubar(I,J)=ZERO
462             Vbar(I,J)=ZERO
463             U(I,J)=ZERO
464             V(I,J)=ZERO
465             HU(I,J)=ZERO
466             HV(I,J)=ZERO
467            ELSE
468             HU(I,J)=Max(H(I,J),MinDepthFrc)*U(I,J)
469             HV(I,J)=Max(H(I,J),MinDepthFrc)*V(I,J)
470     ! apply Froude cap
471             Utotal=SQRT(U(I,J)*U(I,J)+V(I,J)*V(I,J))
472             Fr=SQRT(GRAV*Max(H(I,J),MinDepthFrc))
473             IF(Utotal/Fr.gt.FroudeCap)THEN
474               Utheta=ATAN2(V(I,J),U(I,J))
475               U(I,J)=FroudeCap*Fr*COS(Utheta)
476               V(I,J)=FroudeCap*Fr*SIN(Utheta)
477               HU(I,J)=U(I,J)*Max(H(I,J),MinDepthFrc)
478               HV(I,J)=V(I,J)*Max(H(I,J),MinDepthFrc)
479             ENDIF
480     ! end Froude cap
481            ENDIF
482          ENDDO
483          ENDDO
484     
485     !------------ykchoi 07/26/2016
486     !   --- is this for after correction of Froude cap?
487          IF( .NOT. DISPERSION) THEN 			
488     
489     	  Ubar = HU
490     	  Vbar = HV
491     
492          ENDIF
493     
494     END SUBROUTINE GET_Eta_U_V_HU_HV
