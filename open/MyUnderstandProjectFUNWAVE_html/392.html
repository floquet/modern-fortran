56          USE GLOBAL
57     # if defined(INTEL)
58          USE IFPORT
59     # endif
60          IMPLICIT NONE
61     
62     ! internal wavemaker of wei and kirby
63     
64          IF(WaveMaker(1:6)=='WK_REG') THEN
65     
66     ! Reniel complaint about regular wave case with wrong Yc
67     ! it turns out this part is a redefinition of Yc_WK, remove it (07/08/2016)
68     
69     !# if defined (PARALLEL)
70     !# if defined (CARTESIAN)
71     !     Yc_WK=(INT((Nloc-1)/2)+Nghost)*DY
72     !# else
73     !     Yc_WK=(INT((Nloc-1)/2)+Nghost)*DY(1,1)
74     !# endif
75     !# else
76     !# if defined (CARTESIAN)
77     !     Yc_WK=INT(Nglob/2)*DY
78     !# else
79     !     Yc_WK=INT(Nglob/2)*DY(1,1)
80     !# endif
81     !# endif
82         
83          IF(PERIODIC)THEN
84     
85         ! calculate wave number
86            tmp1 = -0.39_SP + 1.0_SP / 3.0_SP  ! alpha1
87            tmp2 = 2.*pi/Tperiod               ! omgn
88            tmp2 = tmp2*tmp2*DEP_WK/grav       ! tb
89            tmp3 = 1.0_SP + tmp2*(-0.39_SP)    ! tc
90     
91           IF(DEP_WK==ZERO.OR.Tperiod==ZERO)THEN
92     # if defined (PARALLEL)
93              if(myid.eq.0) write(*,*) 're-set depth, Tperiod for wavemaker, STOP!'
94              call MPI_FINALIZE ( ier )
95     # else
96              WRITE(*,*)'re-set depth, Tperiod for wavemaker, STOP!'
97              STOP
98     # endif
99           ELSE       
100            tmp1 = SQRT((tmp3-SQRT(tmp3*tmp3-4.0_SP*tmp1*tmp2))  &
101                     /(2.0_SP*tmp1))/DEP_WK     ! wkn 
102           ENDIF 
103          IF(Theta_WK.NE.ZERO)THEN 
104           IF(Theta_WK.GT.ZERO)THEN   
105            tmp3=ZERO
106            I=0
107            Do WHILE (tmp3<Theta_WK)
108              I=I+1
109     !         tmp2=I*2.0_SP*pi/DY/(Jend-Jbeg)     ! rlamda
110     # if defined (CARTESIAN)
111               tmp2=I*2.0_SP*pi/DY/(Nglob-1.0_SP)
112     # else
113               tmp2=I*2.0_SP*pi/DY(1,1)/(Nglob-1.0_SP)
114     # endif
115              IF(tmp2.GE.tmp1)THEN
116               tmp3=90.0
117     
118     # if defined (PARALLEL)
119              if(myid.eq.0) write(*,*) 'should enlarge domain for periodic boundary with this wave angle, STOP'
120              call MPI_FINALIZE ( ier )
121     # else
122               WRITE(*,*)'should enlarge domain for periodic boundary with this wave angle, STOP'
123               STOP
124     # endif
125              ELSE
126                tmp3=ASIN(tmp2/tmp1)*180.0_SP/pi    ! theta, based on rlamda=wkn*sin(theta)
127              ENDIF
128              IF(I>1000)THEN
129     # if defined (PARALLEL)
130              if(myid.eq.0) write(*,*)'could not find a wave angle for periodic boundary condition, STOP'
131                call MPI_FINALIZE ( ier )
132     # else
133                WRITE(*,*) 'could not find a wave angle for periodic boundary condition, STOP'
134                STOP
135     # endif
136              ENDIF
137            ENDDO
138           ELSEIF(Theta_WK.LT.ZERO)THEN
139            tmp3=ZERO
140            I=0
141            Do WHILE (tmp3>Theta_WK)
142              I=I+1
143     !         tmp2=I*2.0_SP*pi/DY/(Jend-Jbeg)     ! rlamda
144     # if defined (CARTESIAN)
145              tmp2=I*2.0_SP*pi/DY/(Nglob-1.0_SP)     ! rlamda
146     # else
147              tmp2=I*2.0_SP*pi/DY(1,1)/(Nglob-1.0_SP)     ! rlamda
148     # endif
149              IF(tmp2.GE.tmp1)THEN
150               tmp3=-90.0
151     # if defined (PARALLEL)
152              if(myid.eq.0) write(*,*)'should enlarge domain for periodic boundary with this wave angle, STOP'
153                call MPI_FINALIZE ( ier )
154     # else
155               WRITE(*,*)'should enlarge domain for periodic boundary with this wave angle, STOP'
156               STOP
157     # endif
158              ELSE
159                tmp3=-ASIN(tmp2/tmp1)*180.0_SP/pi    ! theta, based on rlamda=wkn*sin(theta)
160              ENDIF
161              IF(I>1000)THEN
162     # if defined (PARALLEL)
163              if(myid.eq.0) write(*,*) 'could not find a wave angle for periodic boundary condition, STOP'
164                call MPI_FINALIZE ( ier )
165     # else
166                WRITE(*,*) 'could not find a wave angle for periodic boundary condition, STOP'
167                STOP
168     # endif
169              ENDIF
170            ENDDO
171           ENDIF
172     
173     # if defined (PARALLEL)
174              if(myid.eq.0)then
175               WRITE(*,*) 'wave angle you set:', Theta_WK
176               WRITE(*,*) 'wave angle in calculation to make periodic boundary:', tmp3
177              endif
178     # else
179            WRITE(*,*) 'wave angle you set:', Theta_WK
180            WRITE(*,*) 'wave angle in calculation to make periodic boundary:', tmp3
181     # endif
182        
183     
184            Theta_WK = tmp3
185          ENDIF
186         ENDIF ! end theta .ne.zero
187     
188            CALL WK_WAVEMAKER_REGULAR_WAVE & 
189                    (Tperiod,AMP_WK,Theta_WK,DEP_WK,Delta_WK,D_gen,rlamda,beta_gen,Width_WK)
190     
191          ENDIF
192          
193          IF(WaveMaker(1:9)=='WK_DATA2D')THEN
194           OPEN(1,FILE=TRIM(WaveCompFile))
195            READ(1,*)NumFreq,NumDir
196            ALLOCATE (WAVE_COMP(NumFreq,NumDir),Beta_gen2D(NumFreq,NumDir),D_gen2D(NumFreq,NumDir),  &
197               Phase2D(NumFreq,NumDir), &
198               Freq(NumFreq),Dire(NumDir),rlamda2D(NumFreq,NumDir))
199     
200            ! define cm sm here to make consistence with irregular wave
201            ALLOCATE (Cm(Mloc,Nloc,NumFreq),Sm(Mloc,Nloc,NumFreq))
202            ALLOCATE (OMGN2D(NumFreq))
203      
204            READ(1,*)PeakPeriod
205            DO J=1,NumFreq
206               READ(1,*)Freq(J)
207            ENDDO
208            DO I=1,NumDir
209               READ(1,*)Dire(I)
210            ENDDO
211            DO I=1,NumDir
212              READ(1,*)(WAVE_COMP(J,I),J=1,NumFreq)
213            ENDDO
214     
215            CALL WK_WAVEMAKER_2D_SPECTRAL_DATA & 
216                    (NumFreq,NumDir,Freq,Dire,WAVE_COMP,PeakPeriod,DEP_WK,Delta_WK,D_gen2D,beta_gen2D,&
217                    rlamda2D,Width_WK)
218                    
219     ! random phase
220            DO J=1,NumFreq
221            DO I=1,NumDir
222     # if defined (INTEL)
223               Phase2D(J,I)=rand()*2.0_SP*pi
224     # else
225               Phase2D(J,I)=rand(0)*2.0_SP*pi
226     # endif
227            ENDDO
228            ENDDO
229     
230     
231     !      make efficient calculation fyshi 07/07/2016
232     
233            DO J=1,NumFreq
234               OMGN2D(J) = 2.0_SP*PI*Freq(J)
235            ENDDO
236     
237           FreqPeak = 1.0_SP/PeakPeriod
238     
239           CALL CALCULATE_Cm_Sm(Mloc,Nloc,DX,DY,Xc_WK,Ibeg,Jbeg,NumFreq,NumDir,&
240                    D_gen2D,Phase2D,Width_WK,rlamda2D,beta_gen2D,Cm,Sm)
241     
242           IF(SHOW_BREAKING)THEN
243            T_brk=WAVE_COMP(NumWaveComp,1) 
244           ENDIF
245     
246     !  cannot deallocate phase2d, I could not find any reason why cannot.
247     !  Phase2D is only used here, deallocation will cause minor discontinuity 
248     !  at processor interface
249     !  leave it now 06/10/2016
250     
251            DEALLOCATE (WAVE_COMP,Beta_gen2D,D_gen2D,  &
252     !          Phase2D, &
253               Freq,Dire,rlamda2D)
254     
255     
256          ENDIF ! end wk_data2d
257            
258          IF(WaveMaker(1:7)=='WK_TIME')THEN
259           OPEN(1,FILE=TRIM(WaveCompFile))
260            DO J=1,NumWaveComp
261              READ(1,*)(WAVE_COMP(J,I),I=1,3)
262            ENDDO
263     
264     
265            CALL WK_WAVEMAKER_TIME_SERIES &
266                    (NumWaveComp,WAVE_COMP,PeakPeriod,DEP_WK,Delta_WK,D_genS,beta_genS,Width_WK)
267     
268           IF(SHOW_BREAKING)THEN
269            T_brk=WAVE_COMP(NumWaveComp,1) 
270           ENDIF
271     
272          ENDIF 
273     
274     ! wei and kirby, 1999, irregular wavemaker
275           
276          IF(WaveMaker(1:6)=='WK_IRR') THEN
277     
278     !        move allocation here from init.F 06/07/2016
279     
280            ALLOCATE(D_gen_ir(Nfreq,Ntheta),rlamda_ir(Nfreq,Ntheta), &
281                     phase_ir(Nfreq,Ntheta),&
282                     Beta_gen_ir(Nfreq),omgn_ir(Nfreq), &
283                     Cm(Mloc,Nloc,Nfreq),Sm(Mloc,Nloc,Nfreq))
284     
285     # if defined (CARTESIAN)
286           CALL WK_WAVEMAKER_IRREGULAR_WAVE & 
287            (Nfreq,Ntheta,delta_WK,DEP_WK,FreqPeak,FreqMax,FreqMin,GammaTMA,Hmo,ThetaPeak, &
288              sigma_theta,rlamda_ir,beta_gen_ir,D_gen_ir,Phase_ir,Width_WK,omgn_ir,&
289              Periodic,DY,Nglob)
290           CALL CALCULATE_Cm_Sm(Mloc,Nloc,DX,DY,Xc_WK,Ibeg,Jbeg,Nfreq,Ntheta,&
291                    D_gen_ir,Phase_ir,Width_WK,rlamda_ir,beta_gen_ir,Cm,Sm)
292     # else
293           CALL WK_WAVEMAKER_IRREGULAR_WAVE & 
294            (Nfreq,Ntheta,delta_WK,DEP_WK,FreqPeak,FreqMax,FreqMin,GammaTMA,Hmo,ThetaPeak, &
295              sigma_theta,rlamda_ir,beta_gen_ir,D_gen_ir,Phase_ir,Width_WK,omgn_ir,&
296              Periodic,DY(1,1),Nglob)
297           CALL CALCULATE_Cm_Sm(Mloc,Nloc,DX(1,1),DY(1,1),Xc_WK,Ibeg,Jbeg,Nfreq,Ntheta,&
298                    D_gen_ir,Phase_ir,Width_WK,rlamda_ir,beta_gen_ir,Cm,Sm)
299     # endif
300     
301            ! 06/07/2016 fyshi
302            DEALLOCATE(D_gen_ir,rlamda_ir,phase_ir,&
303                     Beta_gen_ir)
304     
305           IF(SHOW_BREAKING)THEN
306            T_brk=1.0_SP/FreqMax
307           ENDIF
308     
309          ENDIF
310     !   now include spherical 
311     
312     
313          IF(WAVEMAKER(1:3)=='ABS')THEN
314           IF(WAVE_DATA)THEN
315             CALL CALCULATE_DATA2D_Cm_Sm
316           ELSE
317             CALL CALCULATE_TMA_Cm_Sm &
318          (Nfreq,Ntheta,DEP_Ser,FreqPeak,FreqMax,FreqMin,GammaTMA,Hmo,ThetaPeak, &
319              sigma_theta)
320           ENDIF
321          ENDIF
322     
323     END SUBROUTINE WAVEMAKER_INITIALIZATION

326     !-------------------------------------------------------------------------------------
327     !
328     !    WK_WAVEMAKER_2D_SPECTRAL_DATA is subroutine 
329     !     to generate directional spectrum for given 2D directional spectrum
330     !     for given 2D directional spectrum using
331     !     source function for Wei and Kirbys internal wave maker
332     !
333     !    HISTORY: 
334     !      10/17/2011 Fengyan Shi
335     !
336     !-------------------------------------------------------------------------------------
337     SUBROUTINE WK_WAVEMAKER_2D_SPECTRAL_DATA & 
338                    (NumFreq,NumDir,Freq,DireD,WAVE_COMP,PeakPeriod,H_gen,delta,D_gen,beta_gen,rlamda,width)
339          USE PARAM
340          USE GLOBAL, ONLY : PERIODIC,DY,Nglob
341          IMPLICIT NONE
342          INTEGER,INTENT(IN) :: NumFreq,NumDir
343          REAL(SP) :: alpha,alpha1,omgn,tb,tc,wkn,C_phase,wave_length,&
344                      rl_gen,rI,theta,Tperiod,AMP_WK,omgn_tmp
345          REAL(SP),DIMENSION(NumFreq,NumDir), INTENT(OUT) :: D_gen,Beta_gen,rlamda
346          REAL(SP),DIMENSION(NumFreq,NumDir) :: Dir2D
347          REAL(SP),INTENT(OUT) :: width
348          REAL(SP),DIMENSION(NumFreq),INTENT(IN) :: Freq
349          REAL(SP),DIMENSION(NumDir),INTENT(IN) :: DireD
350          REAL(SP),DIMENSION(NumDir) :: Dire
351          REAL(SP),DIMENSION(NumFreq,NumDir),INTENT(IN) :: WAVE_COMP
352          REAL(SP),INTENT(IN) :: H_gen,delta,PeakPeriod
353          INTEGER :: nfre,ndir
354          REAL(SP) :: angle1,angle2
355     
356          Dire=DireD*DEG2RAD
357     
358     ! reorganize direction
359     
360          IF(PERIODIC)THEN
361     
362           DO nfre=1,NumFreq
363           DO ndir=1,NumDir
364     
365            tmp1 = -0.39_SP + 1.0_SP / 3.0_SP  ! alpha1
366            tmp2 = 2.*pi*Freq(nfre)               ! omgn
367            tmp2 = tmp2*tmp2*H_gen/grav       ! tb
368            tmp3 = 1.0_SP + tmp2*(-0.39_SP)    ! tc
369         
370            tmp1 = SQRT((tmp3-SQRT(tmp3*tmp3-4.0_SP*tmp1*tmp2))  &
371                     /(2.0_SP*tmp1))/MAX(SMALL,H_gen)     ! wkn 
372        
373          IF(Dire(ndir).NE.ZERO)THEN 
374           IF(Dire(ndir).GT.ZERO)THEN   
375            tmp3=ZERO
376            I=0
377            Do WHILE (tmp3<Dire(ndir))
378              I=I+1
379     # if defined (CARTESIAN)
380              tmp2=I*2.0_SP*pi/DY/(Nglob-1.0_SP)     ! rlamda
381     # else
382              tmp2=I*2.0_SP*pi/DY(1,1)/(Nglob-1.0_SP)     ! rlamda
383     # endif
384              IF(tmp2.GE.tmp1)THEN
385               tmp3=pi*0.5_SP-SMALL
386              ELSE
387                tmp3=ASIN(tmp2/tmp1)   ! theta, based on rlamda=wkn*sin(theta)
388              ENDIF
389              IF(I>1000)THEN
390                tmp3=pi*0.5_SP-SMALL
391              ENDIF
392            ENDDO
393     
394     ! judge between I-1 and I which is closer
395     # if defined (CARTESIAN)
396              angle1=ASIN((I-1)*2.0_SP*pi/DY/(Nglob-1.0_SP)/tmp1)
397     # else
398              angle1=ASIN((I-1)*2.0_SP*pi/DY(I,J)/(Nglob-1.0_SP)/tmp1)
399     # endif
400              if (abs(angle1-Dire(ndir))<abs(Dire(ndir)-tmp3))then
401                 angle2=angle1
402              else
403                 angle2=tmp3
404              endif
405     
406           ELSEIF(Dire(ndir).LT.ZERO)THEN
407            tmp3=ZERO
408            I=0
409            Do WHILE (tmp3>Dire(ndir))
410              I=I+1
411     # if defined (CARTESIAN)
412              tmp2=I*2.0_SP*pi/DY/(Nglob-1.0_SP)     ! rlamda
413     # else
414              tmp2=I*2.0_SP*pi/DY(I,J)/(Nglob-1.0_SP)     ! rlamda
415     # endif
416              IF(tmp2.GE.tmp1)THEN
417               tmp3=-pi*0.5_SP+SMALL
418              ELSE
419                tmp3=-ASIN(tmp2/tmp1)   ! theta, based on rlamda=wkn*sin(theta)
420              ENDIF
421              IF(I>1000)THEN
422                tmp3=-pi*0.5_SP+SMALL
423              ENDIF
424            ENDDO
425     
426     ! judge between I-1 and I which is closer
427     # if defined (CARTESIAN)
428              angle1=-ASIN((I-1)*2.0_SP*pi/DY/(Nglob-1.0_SP)/tmp1)
429     # else
430              angle1=-ASIN((I-1)*2.0_SP*pi/DY(1,1)/(Nglob-1.0_SP)/tmp1)
431     # endif
432     
433              if (abs(angle1-Dire(ndir))<abs(Dire(ndir)-tmp3))then
434                 angle2=angle1
435              else
436                 angle2=tmp3
437              endif
438           ENDIF 
439     
440     # if defined (PARALLEL)
441     # else
442     
443            WRITE(*,*) 'wave angle you set:', Dire(ndir)*180/pi
444            WRITE(*,*) 'wave angle in calculation:', angle2*180/pi
445            WRITE(*,*) 'chosen between', angle1*180/pi, 'and', tmp3*180/pi
446     
447            WRITE(3,*) 'wave angle you set:', Dire(ndir)*180/pi
448            WRITE(3,*) 'wave angle in calculation:', angle2*180/pi
449            WRITE(3,*) 'chosen between', angle1*180/pi, 'and', tmp3*180/pi
450     # endif
451     
452            Dir2D(nfre,ndir) = angle2
453     
454         ENDIF ! end theta .ne.zero
455     
456         ENDDO
457         ENDDO
458     
459         ELSE ! no periodic
460     
461            DO ndir=1,NumDir
462            DO nfre=1,NumFreq
463              Dir2D(nfre,ndir)=Dire(ndir)
464            ENDDO
465            ENDDO
466     
467         ENDIF ! end periodic
468     
469             alpha=-0.39_SP
470             alpha1=alpha+1.0_SP/3.0_SP
471     
472           DO ndir=1,NumDir
473            DO nfre=1,NumFreq
474     
475             theta = Dir2D(nfre,ndir) 
476     
477             omgn=2.*pi*Freq(nfre)
478             Tperiod = 1.0_SP/Freq(nfre)
479             AMP_WK = WAVE_COMP(nfre,ndir)
480     
481             tb=omgn*omgn*h_gen/grav
482             tc=1.+tb*alpha
483             IF(h_gen==ZERO.OR.Tperiod==ZERO)THEN
484              WRITE(*,*)'re-set depth, Tperiod for wavemaker, STOP!'
485              STOP
486             ELSE
487               wkn=SQRT((tc-SQRT(tc*tc-4.0_SP*alpha1*tb))  &
488                     /(2.0_SP*alpha1))/h_gen
489               C_phase=1./wkn/Tperiod*2.*pi
490             ENDIF
491             wave_length=C_phase*Tperiod
492     
493             rlamda(nfre,ndir)=wkn*sin(theta)
494             beta_gen(nfre,ndir)=80.0_SP/delta**2/wave_length**2
495             rl_gen=wkn*cos(theta)
496             rI=SQRT(3.14159/beta_gen(nfre,ndir))*exp(-rl_gen**2/4./beta_gen(nfre,ndir))
497     
498             D_gen(nfre,ndir)=2.0_SP*AMP_WK  &
499                 *cos(theta)*(omgn**2-alpha1*grav*wkn**4*h_gen**3) &
500                 /(omgn*wkn*rI*(1.0_SP-alpha*(wkn*h_gen)**2))
501     
502              ENDDO
503            ENDDO
504     
505     ! calculate width
506             omgn_tmp=2.0_SP*pi/PeakPeriod
507             tb=omgn_tmp*omgn_tmp*h_gen/grav
508             tc=1.0_SP+tb*alpha
509             wkn=SQRT((tc-SQRT(tc*tc-4.0_SP*alpha1*tb))/(2.0_SP*alpha1))/h_gen
510             C_phase=1.0_SP/wkn/PeakPeriod*2.0_SP*pi
511             wave_length=C_phase*PeakPeriod
512             width=delta*wave_length/2.0_SP
513     
514     END SUBROUTINE WK_WAVEMAKER_2D_SPECTRAL_DATA

516     !-------------------------------------------------------------------------------------
517     !
518     !    CALCULATE_Cm_Sm is subroutine 
519     !     to calculate Cm Sm for Wei and Kirbys 
520     !     internal wave maker, irregular wave (TMA)
521     !
522     !    HISTORY: 
523     !      11/9/2010 Fengyan Shi
524     !
525     !-------------------------------------------------------------------------------------
526     SUBROUTINE CALCULATE_Cm_Sm(M,N,DX,DY,Xc,Ibeg,Jbeg,mfreq,mtheta,D_gen,phi1, &
527                    width,rlamda,beta_gen,Cm,Sm)
528          USE PARAM
529     # if defined (PARALLEL)
530          USE GLOBAL, ONLY : myid,npx,npy,px,py,Mglob,Nglob
531     # endif
532          IMPLICIT NONE
533          INTEGER,INTENT(IN) :: M,N,mfreq,mtheta,Ibeg,Jbeg    
534          REAL(SP),INTENT(IN) :: DX,DY,width,Xc
535          REAL(SP),DIMENSION(mfreq,mtheta),INTENT(IN) :: D_gen,phi1,rlamda 
536          REAL(SP),DIMENSION(mfreq),INTENT(IN) :: beta_gen
537          REAL(SP),DIMENSION(M,N,mfreq),INTENT(OUT) :: Cm,Sm
538          INTEGER::kf,ktheta
539     
540     
541             Cm=ZERO
542             Sm=ZERO
543             DO J=1,N
544             DO I=1,M
545               do kf=1,mfreq
546                do ktheta=1,mtheta
547     
548     # if defined (PARALLEL)
549                 Cm(i,j,kf)=Cm(i,j,kf) &
550                  +D_gen(kf,ktheta)*exp(-beta_gen(kf)*((I-Ibeg +npx*Mglob/px)*DX-Xc)**2) &
551               *cos(rlamda(kf,ktheta) &
552               *((J-Jbeg  +npy*Nglob/py)*DY-ZERO)+phi1(kf,ktheta))
553     
554                 Sm(i,j,kf)=Sm(i,j,kf) &
555                  +D_gen(kf,ktheta)*exp(-beta_gen(kf)*((I-Ibeg+ npx*Mglob/px)*DX-Xc)**2) &
