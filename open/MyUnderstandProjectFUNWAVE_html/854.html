48     !    COUPLING_INITIAL is the subroutine to initialize coupling/nesting conditions
49     !    
50     !    HISTORY: 05/01/2010 Fengyan Shi
51     !
52     !-------------------------------------------------------------------------------------
53     SUBROUTINE COUPLING_INITIAL
54         USE GLOBAL
55         USE INPUT_READ
56         IMPLICIT NONE   
57       
58      INQUIRE(FILE=TRIM(COUPLING_FILE),EXIST=FILE_EXIST)
59       IF(.NOT.FILE_EXIST)THEN
60     # if defined (PARALLEL)
61        IF(MYID==0)  &
62        WRITE(*,*) TRIM(COUPLING_FILE), ' specified in input.txt but CANNOT BE FOUND. STOP'
63        CALL MPI_FINALIZE (ier)
64        STOP
65     # else
66         WRITE(*,*) TRIM(COUPLING_FILE), ' specified in input.txt but CANNOT BE FOUND. STOP'
67         STOP
68     # endif
69       ENDIF
70     
71            OPEN(11,FILE=TRIM(COUPLING_FILE))
72     
73     
74              READ(11,*)  ! title
75              READ(11,*)  ! boundary info
76     ! boundary basic info including point number of coupling, start point, etc
77     ! east
78              READ(11,*)  ! east
79              READ(11,*) N_COUPLING_EAST,J_START_EAST
80     ! west 
81              READ(11,*)  ! west
82              READ(11,*) N_COUPLING_WEST,J_START_WEST
83     ! south 
84              READ(11,*)  ! south
85              READ(11,*) N_COUPLING_SOUTH,I_START_SOUTH
86     ! north 
87              READ(11,*)  ! north
88              READ(11,*) N_COUPLING_NORTH,I_START_NORTH
89     
90     ! read time and variable at the first level
91     
92              READ(11,*) ! time start title
93              READ(11,*) TIME_COUPLING_1 
94     ! initialize time_2
95              TIME_COUPLING_2 = TIME_COUPLING_1
96     
97     ! east
98              IF(N_COUPLING_EAST.GT.0)THEN
99                ALLOCATE(U_COUPLING_EAST(N_COUPLING_EAST,2),&
100                    V_COUPLING_EAST(N_COUPLING_EAST,2),&
101                    Z_COUPLING_EAST(N_COUPLING_EAST,2))
102                  READ(11,*)   ! east
103                  READ(11,119)(U_COUPLING_EAST(I,2),I=1,N_COUPLING_EAST)
104                  READ(11,119)(V_COUPLING_EAST(I,2),I=1,N_COUPLING_EAST)
105                  READ(11,119)(Z_COUPLING_EAST(I,2),I=1,N_COUPLING_EAST)
106     !   initialize first step
107                  U_COUPLING_EAST(:,1)=U_COUPLING_EAST(:,2)
108                  V_COUPLING_EAST(:,1)=V_COUPLING_EAST(:,2)
109                  Z_COUPLING_EAST(:,1)=Z_COUPLING_EAST(:,2)
110              ELSE
111                  READ(11,*)
112     
113              ENDIF ! n_coupling_east
114     119      FORMAT(5E16.6)
115     
116     ! west
117              IF(N_COUPLING_WEST.GT.0)THEN
118                ALLOCATE(U_COUPLING_WEST(N_COUPLING_WEST,2),&
119                    V_COUPLING_WEST(N_COUPLING_WEST,2),&
120                    Z_COUPLING_WEST(N_COUPLING_WEST,2))
121                  READ(11,*)   ! west
122                  READ(11,119)(U_COUPLING_WEST(I,2),I=1,N_COUPLING_WEST)
123                  READ(11,119)(V_COUPLING_WEST(I,2),I=1,N_COUPLING_WEST)
124                  READ(11,119)(Z_COUPLING_WEST(I,2),I=1,N_COUPLING_WEST)
125     !   initialize first step
126                  U_COUPLING_WEST(:,1)=U_COUPLING_WEST(:,2)
127                  V_COUPLING_WEST(:,1)=V_COUPLING_WEST(:,2)
128                  Z_COUPLING_WEST(:,1)=Z_COUPLING_WEST(:,2)
129              ELSE
130                  READ(11,*)
131     
132              ENDIF ! n_coupling_west
133     ! south
134              IF(N_COUPLING_SOUTH.GT.0)THEN
135                ALLOCATE(U_COUPLING_SOUTH(N_COUPLING_SOUTH,2),&
136                    V_COUPLING_SOUTH(N_COUPLING_SOUTH,2),&
137                    Z_COUPLING_SOUTH(N_COUPLING_SOUTH,2))
138                  READ(11,*)   ! south
139                  READ(11,119)(U_COUPLING_SOUTH(I,2),I=1,N_COUPLING_SOUTH)
140                  READ(11,119)(V_COUPLING_SOUTH(I,2),I=1,N_COUPLING_SOUTH)
141                  READ(11,119)(Z_COUPLING_SOUTH(I,2),I=1,N_COUPLING_SOUTH)
142     !   initialize first step
143                  U_COUPLING_SOUTH(:,1)=U_COUPLING_SOUTH(:,2)
144                  V_COUPLING_SOUTH(:,1)=V_COUPLING_SOUTH(:,2)
145                  Z_COUPLING_SOUTH(:,1)=Z_COUPLING_SOUTH(:,2)
146              ELSE
147                  READ(11,*)
148     
149              ENDIF ! n_coupling_south
150     ! north
151              IF(N_COUPLING_NORTH.GT.0)THEN
152                ALLOCATE(U_COUPLING_NORTH(N_COUPLING_NORTH,2),&
153                    V_COUPLING_NORTH(N_COUPLING_NORTH,2),&
154                    Z_COUPLING_NORTH(N_COUPLING_NORTH,2))
155                  READ(11,*)   ! north
156                  READ(11,119)(U_COUPLING_NORTH(I,2),I=1,N_COUPLING_NORTH)
157                  READ(11,119)(V_COUPLING_NORTH(I,2),I=1,N_COUPLING_NORTH)
158                  READ(11,119)(Z_COUPLING_NORTH(I,2),I=1,N_COUPLING_NORTH)
159     
160     !   initialize first step
161                  U_COUPLING_NORTH(:,1)=U_COUPLING_NORTH(:,2)
162                  V_COUPLING_NORTH(:,1)=V_COUPLING_NORTH(:,2)
163                  Z_COUPLING_NORTH(:,1)=Z_COUPLING_NORTH(:,2)
164              ELSE
165                  READ(11,*)
166     
167              ENDIF ! n_coupling_north
168     
169     
170     ! specify boundary start points
171     
172     ! west boundary
173        IF(N_COUPLING_WEST>0)THEN
174     # if defined (PARALLEL)
175         if ( n_west .eq. MPI_PROC_NULL ) then
176           Kstart_WEST=J_START_WEST+Nghost -npy*Nglob/py
177           Kend_WEST = J_START_WEST+Nghost+N_COUPLING_WEST-1 -npy*Nglob/py
178           IF((Kstart_WEST>Nghost.AND.Kstart_WEST<Nloc-Nghost+1).OR.&
179          (Kend_WEST<Nloc-Nghost+1.AND.Kend_WEST>Nghost))THEN
180            IF(Kstart_WEST<Nghost+1)THEN
181              Kshift_WEST= -Kstart_WEST+Nghost+1
182              Kstart_WEST=Nghost+1
183            ELSE
184              Kshift_WEST=-(Kstart_WEST-Nghost)+1
185            ENDIF
186            IF(Kend_WEST>Nloc-Nghost)THEN
187              Kend_WEST=Nloc-Nghost
188            ENDIF
189            IN_DOMAIN_WEST=.TRUE.
190           ELSE
191            IF(Kstart_WEST<=Nghost.AND.Kend_WEST>=Nloc-Nghost+1)THEN
192              Kshift_WEST=-Kstart_WEST+Nghost+1
193              Kstart_WEST = Nghost+1
194              Kend_WEST = Nloc-Nghost
195              IN_DOMAIN_WEST=.TRUE.
196            ELSE
197              IN_DOMAIN_WEST=.FALSE.
198            ENDIF
199           ENDIF
200     ! check print*,myid,Kshift_WEST,Kstart_WEST,Kend_WEST
201          endif
202     # else
203           Kstart_WEST=J_START_WEST+Nghost
204           Kend_WEST = J_START_WEST+Nghost+N_COUPLING_WEST-1
205           Kshift_WEST = -(Kstart_WEST-Nghost)+1
206           IN_DOMAIN_WEST = .TRUE.
207     # endif
208     
209        ENDIF
210     
211     ! east boundary
212        IF(N_COUPLING_EAST>0)THEN
213     # if defined (PARALLEL)
214         if ( n_east .eq. MPI_PROC_NULL ) then
215           Kstart_EAST=J_START_EAST+Nghost -npy*Nglob/py
216           Kend_EAST = J_START_EAST+Nghost+N_COUPLING_EAST-1 -npy*Nglob/py
217           IF((Kstart_EAST>Nghost.AND.Kstart_EAST<Nloc-Nghost+1).OR.&
218          (Kend_EAST<Nloc-Nghost+1.AND.Kend_EAST>Nghost))THEN
219            IF(Kstart_EAST<Nghost+1)THEN
220              Kshift_EAST= -Kstart_EAST+Nghost+1
221              Kstart_EAST=Nghost+1
222            ELSE
223              Kshift_EAST=-(Kstart_EAST-Nghost)+1
224            ENDIF
225            IF(Kend_EAST>Nloc-Nghost)THEN
226              Kend_EAST=Nloc-Nghost
227            ENDIF
228            IN_DOMAIN_EAST=.TRUE.
229           ELSE
230     
231            IF(Kstart_EAST<=Nghost.AND.Kend_EAST>=Nloc-Nghost+1)THEN
232              Kshift_EAST=-Kstart_EAST+Nghost+1
233              Kstart_EAST = Nghost+1
234              Kend_EAST = Nloc-Nghost
235              IN_DOMAIN_EAST=.TRUE.
236            ELSE
237              IN_DOMAIN_EAST=.FALSE.
238            ENDIF
239           ENDIF
240     
241         endif
242     # else
243           Kstart_EAST=J_START_EAST+Nghost
244           Kend_EAST = J_START_EAST+Nghost+N_COUPLING_EAST-1
245           Kshift_EAST = -(Kstart_EAST-Nghost)+1
246           IN_DOMAIN_EAST = .TRUE.
247     # endif
248         ENDIF
249     
250     ! south boundary
251        IF(N_COUPLING_SOUTH>0)THEN
252     # if defined (PARALLEL)
253         if ( n_suth .eq. MPI_PROC_NULL ) then
254           Kstart_SOUTH=I_START_SOUTH+Nghost -npx*Mglob/px
255           Kend_SOUTH = I_START_SOUTH+Nghost+N_COUPLING_SOUTH-1 -npx*Mglob/px
256           IF((Kstart_SOUTH>Nghost.AND.Kstart_SOUTH<Mloc-Nghost+1).OR.&
257          (Kend_SOUTH<Mloc-Nghost+1.AND.Kend_SOUTH>Nghost))THEN
258            IF(Kstart_SOUTH<Nghost+1)THEN
259              Kshift_SOUTH= -Kstart_SOUTH+Nghost+1
260              Kstart_SOUTH=Nghost+1
261            ELSE
262              Kshift_SOUTH=-(Kstart_SOUTH-Nghost)+1
263            ENDIF
264            IF(Kend_SOUTH>Mloc-Nghost)THEN
265              Kend_SOUTH=Mloc-Nghost
266            ENDIF
267            IN_DOMAIN_SOUTH=.TRUE.
268           ELSE
269     
270            IF(Kstart_SOUTH<=Nghost.AND.Kend_SOUTH>=Mloc-Nghost+1)THEN
271              Kshift_SOUTH=-Kstart_SOUTH+Nghost+1
272              Kstart_SOUTH = Nghost+1
273              Kend_SOUTH = Mloc-Nghost
274              IN_DOMAIN_SOUTH=.TRUE.
275            ELSE
276              IN_DOMAIN_SOUTH=.FALSE.
277            ENDIF
278           ENDIF
279     
280         endif
281     # else
282           Kstart_SOUTH=I_START_SOUTH+Nghost
283           Kend_SOUTH = I_START_SOUTH+Nghost+N_COUPLING_SOUTH-1
284           Kshift_SOUTH = -(Kstart_SOUTH-Nghost)+1
285           IN_DOMAIN_SOUTH = .TRUE.
286     # endif
287        ENDIF
288     
289     ! north boundary
290        IF(N_COUPLING_NORTH>0)THEN
291     # if defined (PARALLEL)
292         if ( n_nrth .eq. MPI_PROC_NULL ) then
293           Kstart_NORTH=I_START_NORTH+Nghost -npx*Mglob/px
294           Kend_NORTH = I_START_NORTH+Nghost+N_COUPLING_NORTH-1 -npx*Mglob/px
295           IF((Kstart_NORTH>Nghost.AND.Kstart_NORTH<Mloc-Nghost+1).OR.&
296          (Kend_NORTH<Mloc-Nghost+1.AND.Kend_NORTH>Nghost))THEN
297            IF(Kstart_NORTH<Nghost+1)THEN
298              Kshift_NORTH= -Kstart_NORTH+Nghost+1
299              Kstart_NORTH=Nghost+1
300            ELSE
301              Kshift_NORTH=-(Kstart_NORTH-Nghost)+1
302            ENDIF
303            IF(Kend_NORTH>Mloc-Nghost)THEN
304              Kend_NORTH=Mloc-Nghost
305            ENDIF
306            IN_DOMAIN_NORTH=.TRUE.
307           ELSE
308     
309            IF(Kstart_NORTH<=Nghost.AND.Kend_NORTH>=Mloc-Nghost+1)THEN
310              Kshift_NORTH=-Kstart_NORTH+Nghost+1
311              Kstart_NORTH = Nghost+1
312              Kend_NORTH = Mloc-Nghost
313              IN_DOMAIN_NORTH=.TRUE.
314            ELSE
315              IN_DOMAIN_NORTH=.FALSE.
316            ENDIF
317           ENDIF
318     
319         endif
320     # else
321           Kstart_NORTH=I_START_NORTH+Nghost
322           Kend_NORTH = I_START_NORTH+Nghost+N_COUPLING_NORTH-1
323           Kshift_NORTH = -(Kstart_NORTH-Nghost)+1
324           IN_DOMAIN_NORTH = .TRUE.
325     # endif
326        ENDIF
327     
328     END SUBROUTINE COUPLING_INITIAL
329     # endif

332     # if defined (COUPLING)
333     !-------------------------------------------------------------------------------------
334     !
335     !   OneWayCoupling subroutine is used to pass coupling variables into ghost cells                                                         
336     !
337     !   HISTORY: 04/28/2012 Fengyan Shi   
338     !   1) used format I/O  2) read through read(11,119)                                      
339     !
340     !-------------------------------------------------------------------------------------
341     SUBROUTINE OneWayCoupling
342         USE GLOBAL
343         IMPLICIT NONE
344     
345     119      FORMAT(5E16.6)  ! this is a fixed format for I/O
346      
347     ! determine time slot
348     
349         IF(TIME>TIME_COUPLING_1.AND.TIME>TIME_COUPLING_2) THEN
350              TIME_COUPLING_1=TIME_COUPLING_2
351     
352              READ(11,*,END=120) TIME_COUPLING_2 
353     ! east
354              IF(N_COUPLING_EAST.GT.0)THEN
355                  READ(11,*,END=120)   ! east
356     
357                  U_COUPLING_EAST(:,1)=U_COUPLING_EAST(:,2)
358                  V_COUPLING_EAST(:,1)=V_COUPLING_EAST(:,2)
359                  Z_COUPLING_EAST(:,1)=Z_COUPLING_EAST(:,2)
360     
361                  READ(11,119,END=120)(U_COUPLING_EAST(I,2),I=1,N_COUPLING_EAST)
362                  READ(11,119,END=120)(V_COUPLING_EAST(I,2),I=1,N_COUPLING_EAST)
363                  READ(11,119,END=120)(Z_COUPLING_EAST(I,2),I=1,N_COUPLING_EAST)
364              ELSE
365                  READ(11,*,END=120)   ! east            
366              ENDIF
367     ! west
368              IF(N_COUPLING_WEST.GT.0)THEN
369                  READ(11,*,END=120)   ! west
370     
371                  U_COUPLING_WEST(:,1)=U_COUPLING_WEST(:,2)
372                  V_COUPLING_WEST(:,1)=V_COUPLING_WEST(:,2)
373                  Z_COUPLING_WEST(:,1)=Z_COUPLING_WEST(:,2)
374     
375     
376                  READ(11,119,END=120)(U_COUPLING_WEST(I,2),I=1,N_COUPLING_WEST)
377                  READ(11,119,END=120)(V_COUPLING_WEST(I,2),I=1,N_COUPLING_WEST)
378                  READ(11,119,END=120)(Z_COUPLING_WEST(I,2),I=1,N_COUPLING_WEST)
379              ELSE
380                  READ(11,*,END=120)   ! west            
381              ENDIF
382     ! south
383              IF(N_COUPLING_SOUTH.GT.0)THEN
384                  READ(11,*,END=120)   ! south
385     
386                  U_COUPLING_SOUTH(:,1)=U_COUPLING_SOUTH(:,2)
387                  V_COUPLING_SOUTH(:,1)=V_COUPLING_SOUTH(:,2)
388                  Z_COUPLING_SOUTH(:,1)=Z_COUPLING_SOUTH(:,2)
389     
390     
391                  READ(11,119,END=120)(U_COUPLING_SOUTH(I,2),I=1,N_COUPLING_SOUTH)
392                  READ(11,119,END=120)(V_COUPLING_SOUTH(I,2),I=1,N_COUPLING_SOUTH)
393                  READ(11,119,END=120)(Z_COUPLING_SOUTH(I,2),I=1,N_COUPLING_SOUTH)
394              ELSE
395                  READ(11,*,END=120)   ! south            
396              ENDIF
397     ! north
398              IF(N_COUPLING_NORTH.GT.0)THEN
399                  READ(11,*,END=120)   ! north
400     
401                  U_COUPLING_NORTH(:,1)=U_COUPLING_NORTH(:,2)
402                  V_COUPLING_NORTH(:,1)=V_COUPLING_NORTH(:,2)
403                  Z_COUPLING_NORTH(:,1)=Z_COUPLING_NORTH(:,2)
404     
405                  READ(11,119,END=120)(U_COUPLING_NORTH(I,2),I=1,N_COUPLING_NORTH)
406                  READ(11,119,END=120)(V_COUPLING_NORTH(I,2),I=1,N_COUPLING_NORTH)
407                  READ(11,119,END=120)(Z_COUPLING_NORTH(I,2),I=1,N_COUPLING_NORTH)
408              ELSE
409                  READ(11,*,END=120)   ! north            
410              ENDIF
411     
412     
413         ENDIF  ! time>time_2 and time_1
414     
415     120 CONTINUE
416     
417         tmp2=ZERO
418         tmp1=ZERO
419     
420         IF(TIME>TIME_COUPLING_1)THEN
421           IF(TIME_COUPLING_1.EQ.TIME_COUPLING_2)THEN
422             ! no more data
423             tmp2=ZERO
424             tmp1=ZERO
425           ELSE
426           tmp2=(TIME_COUPLING_2-TIME) &
427                 /MAX(SMALL, ABS(TIME_COUPLING_2-TIME_COUPLING_1))
428           tmp1=1.0_SP - tmp2;
429           ENDIF  ! no more data?
430         ENDIF ! time>time_1
431     
432     ! west boundary
433        IF(N_COUPLING_WEST>0)THEN
434     # if defined (PARALLEL)
435         if ( n_west .eq. MPI_PROC_NULL ) then
436     # endif
437          IF(IN_DOMAIN_WEST)THEN
438     
439           DO J=Kstart_WEST,Kend_WEST 
440           DO I=1,Nghost
441             ETA(I,J)=Z_COUPLING_WEST(J-Nghost+Kshift_WEST,2)*tmp1&
442                     +Z_COUPLING_WEST(J-Nghost+Kshift_WEST,1)*tmp2
443             U(I,J)=U_COUPLING_WEST(J-Nghost+Kshift_WEST,2)*tmp1&
444                     +U_COUPLING_WEST(J-Nghost+Kshift_WEST,1)*tmp2
445             V(I,J)=V_COUPLING_WEST(J-Nghost+Kshift_WEST,2)*tmp1&
446                     +V_COUPLING_WEST(J-Nghost+Kshift_WEST,1)*tmp2
447             HU(I,J)=(Depth(I,J)+ETA(I,J))*U(I,J)
448             HV(I,J)=(Depth(I,J)+ETA(I,J))*V(I,J)
449           ENDDO
450           ENDDO
451          ENDIF  ! end in domain
452     # if defined (PARALLEL)
453         endif
454     # endif
455         ENDIF ! end of n_coupling_west>0
456     
457     ! east boundary
458        IF(N_COUPLING_EAST>0)THEN
459     # if defined (PARALLEL)
460         if ( n_east .eq. MPI_PROC_NULL ) then
461     # endif
462          IF(IN_DOMAIN_EAST)THEN
463           DO J=Kstart_EAST,Kend_EAST  
464           DO I=Iend+1,Iend+Nghost
465             ETA(I,J)=Z_COUPLING_EAST(J-Nghost+Kshift_EAST,2)*tmp1&
466                     +Z_COUPLING_EAST(J-Nghost+Kshift_EAST,1)*tmp2
467             U(I,J)=U_COUPLING_EAST(J-Nghost+Kshift_EAST,2)*tmp1&
468                     +U_COUPLING_EAST(J-Nghost+Kshift_EAST,1)*tmp2
469             V(I,J)=V_COUPLING_EAST(J-Nghost+Kshift_EAST,2)*tmp1&
470                     +V_COUPLING_EAST(J-Nghost+Kshift_EAST,1)*tmp2
471             HU(I,J)=(Depth(I,J)+ETA(I,J))*U(I,J)
472             HV(I,J)=(Depth(I,J)+ETA(I,J))*V(I,J)
473           ENDDO
474           ENDDO
475          ENDIF  ! end in domain
476     # if defined (PARALLEL)
477         endif
478     # endif
479         ENDIF ! end of n_coupling_east>0
480     
481     ! south boundary
482        IF(N_COUPLING_SOUTH>0)THEN
483     # if defined (PARALLEL)
484         if ( n_suth .eq. MPI_PROC_NULL ) then
485     # endif
486          IF(IN_DOMAIN_SOUTH)THEN
487           DO I=Kstart_SOUTH,Kend_SOUTH  
488           DO J=1,Nghost
489             ETA(I,J)=Z_COUPLING_SOUTH(I-Nghost+Kshift_SOUTH,2)*tmp1&
490                     +Z_COUPLING_SOUTH(I-Nghost+Kshift_SOUTH,1)*tmp2
