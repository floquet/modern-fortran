46     !    SPONGE_DAMPING is subroutine for dampping waves using Larsen-Dancy(1983)
47     !    type sponge layer 
48     !    
49     !    HISTORY: 10/27/2010 Fengyan Shi
50     !
51     !-------------------------------------------------------------------------------------
52     SUBROUTINE SPONGE_DAMPING
53          USE GLOBAL
54          IMPLICIT NONE
55     
56          DO J=1,Nloc
57          DO I=1,Mloc
58           IF(MASK(I,J)>ZERO)THEN
59            ETA(I,J)=ETA(I,J)/SPONGE(I,J)
60           ENDIF
61            U(I,J)=U(I,J)/SPONGE(I,J)
62            V(I,J)=V(I,J)/SPONGE(I,J)
63          ENDDO
64          ENDDO
65     
66     END SUBROUTINE SPONGE_DAMPING

68     !-------------------------------------------------------------------------------------
69     !
70     !    ABSORBING_GENERATING_BC is subroutine for wave generation using Larsen-Dancy(1983)
71     !      type wave-maker
72     !    
73     !    HISTORY: 
74     !      05/01/2014 Fengyan Shi
75     !
76     !-------------------------------------------------------------------------------------
77     SUBROUTINE ABSORBING_GENERATING_BC
78          USE GLOBAL, ONLY : ETA,U,V,SPONGEMAKER,Mloc,Nloc,MASK,I,J,DX,DY,ZERO
79          USE GLOBAL, ONLY : SP,PI,Grav,TIME,NumFreq,Phase_Ser,&
80                            Ibeg,Iend,Jbeg,Jend,&
81                            Segma_Ser,Stokes_Drift_Ser,Cm_eta,Sm_eta,&
82                            Cm_u,Sm_u,Cm_v,Sm_v,SPONGE
83          IMPLICIT NONE
84          INTEGER :: KK,K
85          real(SP),DIMENSION(Mloc,Nloc) :: Ein2D,Din2D
86          real(SP),DIMENSION(Mloc,Nloc) :: Uin2D,Vin2D
87     
88     
89          Ein2D = ZERO
90          Uin2D = ZERO
91          Vin2D = ZERO
92     
93     
94          do j = 1,Nloc
95          do i = 1,Mloc
96           DO KK = 1, NumFreq
97     
98            Ein2D(I,J) =Ein2D(I,J)+Cm_eta(I,J,KK)*COS(pi/2.0_SP+Segma_Ser(KK)*TIME+Phase_Ser(KK)) &
99                                  +Sm_eta(I,J,KK)*SIN(pi/2.0_SP+Segma_Ser(KK)*TIME+Phase_Ser(KK))
100      
101            Uin2D(I,J) =Uin2D(I,J)+Cm_u(I,J,KK)*COS(pi/2.0_SP+Segma_Ser(KK)*TIME+Phase_Ser(KK)) &
102                                  +Sm_u(I,J,KK)*SIN(pi/2.0_SP+Segma_Ser(KK)*TIME+Phase_Ser(KK))
103            Vin2D(I,J) =Vin2D(I,J)+Cm_v(I,J,KK)*COS(pi/2.0_SP+Segma_Ser(KK)*TIME+Phase_Ser(KK)) &
104                                  +Sm_v(I,J,KK)*SIN(pi/2.0_SP+Segma_Ser(KK)*TIME+Phase_Ser(KK))
105            ENDDO 
106     
107          enddo 
108          enddo   
109     
110            DO J=1,Nloc
111            DO I=1,Mloc
112              Ein2D(I,J)=Ein2D(I,J)/SPONGE(I,J)**4
113              Uin2D(I,J)=Uin2D(I,J)/SPONGE(I,J)**4
114              Vin2D(I,J)=Vin2D(I,J)/SPONGE(I,J)**4
115            ENDDO
116            ENDDO
117     
118     
119            do j = 1,Nloc
120            do i = 1,Mloc
121               Eta(i,j) = Ein2D(I,J)+ (Eta(i,j)-Ein2D(I,J))/SpongeMaker(i,j)
122               U(i,j) = Uin2D(I,J)+(U(i,j)-Uin2D(I,J))/SpongeMaker(i,j)
123               V(i,j) = Vin2D(I,J)+(V(i,j)-Vin2D(I,J))/SpongeMaker(i,j)
124            enddo
125            enddo
126     
127     END SUBROUTINE ABSORBING_GENERATING_BC

129     !-------------------------------------------------------------------------------------
130     !    CALCULATE_SPONGE_MAKER is subroutine for calculation of 
131     !      spong layer coefficient
132     !    
133     !    HISTORY: 
134     !      05/01/2014 Fengyan Shi
135     !-------------------------------------------------------------------------------------
136     SUBROUTINE CALCULATE_SPONGE_MAKER(M,N,Nghost,DX,DY,&
137                                 Sponge_west_width, &
138                                 R_sponge,A_sponge,SPONGE)
139          USE PARAM
140     # if defined (PARALLEL)
141          USE GLOBAL, ONLY : n_west, n_east, n_suth, n_nrth,&
142                      px,py,npx,npy,Mglob,Nglob
143     # endif
144          IMPLICIT NONE
145          INTEGER, INTENT(IN)::M,N,Nghost
146     # if defined (CARTESIAN)
147          REAL(SP),INTENT(IN)::DX,DY
148     # else
149          REAL(SP),DIMENSION(M,N),INTENT(IN)::DX,DY
150     # endif
151          REAL(SP),INTENT(IN) :: &
152                               Sponge_west_width, &
153                               R_sponge,A_sponge
154          REAL(SP),DIMENSION(M,N),INTENT(INOUT)::SPONGE
155          REAL(SP)::ri,lim
156          INTEGER::Iwidth
157     
158     
159     # if defined (CARTESIAN)
160          Iwidth=INT(Sponge_west_width/DX)+Nghost
161     # else
162          Iwidth=INT(Sponge_west_width/DX(1,1))+Nghost
163     # endif
164          DO J=1,N
165          DO I=1,Iwidth
166            IF(SPONGE(I,J)>1.0_SP)THEN
167              lim=SPONGE(I,J)
168            ELSE
169              lim=1.0_SP
170            ENDIF
171     # if defined(PARALLEL)
172            ri = R_Sponge**(50*(i+npx*Mglob/px-1)/(Iwidth-1))
173     # else
174            ri=R_sponge**(50*(I-1)/(Iwidth-1))
175     # endif
176            Sponge(i,j)=MAX(A_Sponge**ri,lim)
177          ENDDO
178          ENDDO
179     
180     END SUBROUTINE CALCULATE_SPONGE_MAKER

247     # else

269     # else
270            ri=MAX(0.0,REAL(Iwidth-M+I))
271     # endif
272            tmp_2d_2(i,j)=MAX(CDsponge*TANH(ri/10.0),lim)
273          ENDDO
274          ENDDO
275          ENDIF
276     
277           DO J=1,N
278           DO I=1,M
279             IF(tmp_2d_1(I,J)>tmp_2d_2(I,J)) THEN
280               Sponge(I,J)=tmp_2d_1(I,J)
281             ELSE
282               Sponge(I,J)=tmp_2d_2(I,J)
283             ENDIF
284           ENDDO
285           ENDDO

294     # endif
295          DO I=1,M
296          DO J=1,N

298              lim=SPONGE(I,J)
299            ELSE
300              lim=0.0_SP
301            ENDIF
302     # if defined(PARALLEL)
303            ri=MAX(0.0,REAL(Iwidth-J-npy*Nglob/py))
304     # else
305            ri=MAX(0.0,REAL(Iwidth-J))
306     # endif
307            tmp_2d_1(i,j)=MAX(CDsponge*TANH(ri/10.0),lim)
308          ENDDO
309          ENDDO
310          ENDIF
311     
312     
313     ! north
314     
315          IF(Sponge_north_width>ZERO)THEN
316     # if defined (CARTESIAN)
317          Iwidth=INT(Sponge_north_width/DY)+Nghost
318     # else
319          Iwidth=INT(Sponge_north_width/DY(1,1))+Nghost
320     # endif
321          DO I=1,M
322          DO J=1,N
323            IF(SPONGE(I,J)>0.0_SP)THEN
324              lim=SPONGE(I,J)
325            ELSE
326              lim=0.0_SP
327            ENDIF
328     # if defined(PARALLEL)
329            ri=MAX(0.0,REAL(Iwidth-N+J-(py-npy-1)*Nglob/py))
330     # else
331            ri=MAX(0.0,REAL(Iwidth-N+J))
332     # endif
333            tmp_2d_2(i,j)=MAX(CDsponge*TANH(ri/10.0),lim)
334          ENDDO
335          ENDDO
336          ENDIF
337     
338           DO J=1,N
339           DO I=1,M
340             IF(tmp_2d_1(I,J)>tmp_2d_2(I,J)) THEN

342             ELSE

344             ENDIF
345           ENDDO
346           ENDDO

613            ri = R_Sponge**(50*(M-i+(px-npx-1)*Mglob/px)/(Iwidth-1))
614     # else
615            ri=R_sponge**(50*(M-I)/(Iwidth-1))
616     # endif
617            tmp_2d_2(i,j)=MAX(A_Sponge**ri,lim)
618          ENDDO
619          ENDDO
620          ENDIF
621     
622           DO J=1,N
623           DO I=1,M
624             IF(tmp_2d_1(I,J)>tmp_2d_2(I,J)) THEN
625               Sponge(I,J)=tmp_2d_1(I,J)
626             ELSE
627               Sponge(I,J)=tmp_2d_2(I,J)
628             ENDIF
629             IF(Sponge(I,J)<A_Sponge**(R_Sponge**50))Sponge(I,J)=1.0_SP
630           ENDDO
631           ENDDO
632     
633     ! south
634     
635          IF(Sponge_south_width>ZERO)THEN
636     # if defined (CARTESIAN)
637          Iwidth=INT(Sponge_south_width/DY)+Nghost
638     # else
639          Iwidth=INT(Sponge_south_width/DY(1,1))+Nghost
640     # endif
641          DO I=1,M
642          DO J=1,N
643            IF(SPONGE(I,J)>1.0_SP)THEN
644              lim=SPONGE(I,J)
645            ELSE
646              lim=1.0_SP
647            ENDIF
648     # if defined(PARALLEL)
649            ri=R_sponge**(50*(J+npy*Nglob/py-1)/(Iwidth-1))
650     # else
651            ri=R_sponge**(50*(J-1)/(Iwidth-1))
652     # endif
653            tmp_2d_1(i,j)=MAX(A_Sponge**ri,lim)
654          ENDDO
655          ENDDO
656          ENDIF
657     
658     
659     ! north
660     
661          IF(Sponge_north_width>ZERO)THEN
662     # if defined (CARTESIAN)
663          Iwidth=INT(Sponge_north_width/DY)+Nghost
664     # else
665          Iwidth=INT(Sponge_north_width/DY(1,1))+Nghost
666     # endif
667          DO I=1,M
668          DO J=1,N
669            IF(SPONGE(I,J)>1.0_SP)THEN
670              lim=SPONGE(I,J)
671            ELSE
672              lim=1.0_SP
673            ENDIF
674     # if defined(PARALLEL)
675            ri = R_Sponge**(50*(N-J+(py-npy-1)*Nglob/py)/(Iwidth-1))
676     # else
677            ri=R_sponge**(50*(N-J)/(Iwidth-1))	  
678     # endif
679            tmp_2d_2(i,j)=MAX(A_Sponge**ri,lim)	  
680          ENDDO
681          ENDDO
682          ENDIF
683     
684           DO J=1,N
685           DO I=1,M
686             IF(tmp_2d_1(I,J)>tmp_2d_2(I,J)) THEN
687               Sponge(I,J)=tmp_2d_1(I,J)
688             ELSE
689               Sponge(I,J)=tmp_2d_2(I,J)
690             ENDIF
691             IF(Sponge(I,J)<A_Sponge**(R_Sponge**50))Sponge(I,J)=1.0_SP	  
692           ENDDO
693           ENDDO
694     
695     END SUBROUTINE CALCULATE_SPONGE
