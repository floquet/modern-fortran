64     !             - SourceTerms again

68     !          ENDDO LOOP
69     !-------------------------------------------------------------------------------------
70     ! ** OPEN FILES **
71     !  (1): read input, (2): output, (3): log, (4): !write/read hotstart
72     !-------------------------------------------------------------------------------------
73     ! ** HOT START DATA **
74     !   NOTE: read input.txt first, if HOT_START, then read  
75             ! -- dimension
76     ! Mloc,Nloc,Mloc1,Nloc1
77     ! Nghost
78     ! Ibeg,Iend,Jbeg,Jend,Iend1,Jend1
79     !   NOTE: need to confirm if the saved data is consistent with input.txt
80             ! -- time
81     ! TIME
82     ! TOTAL_TIME
83     ! PLOT_INTV
84     ! PLOT_COUNT
85     ! SCREEN_INTV
86     ! SCREEN_COUNT
87     ! HOTSTART_INTV
88     ! ICOUNT
89             ! spacing
90     ! DX,DY
91             ! -- physics
92     ! DISPERSION
93     ! Gamma1
94     ! a1,a2,b1,b2
95     ! SWE_ETA_DEP
96             ! -- numerics
97     ! Time_Scheme
98     ! HIGH_ORDER
99     ! CONSTR
100     ! CFL
101     ! FroudeCap
102     ! DISP_TIME_LEFT
103             ! -- wet-dry
104     ! MinDepth,MinDepthfrc
105     
106             ! -- depth
107     ! DEPTH
108     ! DEPTHx
109     ! DEPTHy
110             ! variables
111     ! U
112     ! V
113     ! if (.NOT.DISP_TIME_LEFT)THEN
114     ! U0
115     ! V0
116     ! endif
117     ! Ubar
118     ! Vbar
119     ! ETA 
120     ! H
121     ! MASK
122     ! MASK9
123     ! MAST_STRUC
124     !
125            ! -- wavemaker
126     ! if (WAVEMAKER is WK_IRR)
127     ! turns out the data for Cm Sm too large, calculate it when hotstart
128     !
129     ! if (WAVEMAKER is WK_REG)
130     ! D_gen
131     ! Beta_gen
132     ! rlamda
133     ! 
134     !
135     !-------------------------------------------------------------------------------------
136          USE GLOBAL
137     # if defined (VESSEL)
138          USE VESSEL_MODULE
139     # endif
140     
141          IMPLICIT NONE
142     
143          INTEGER::ISTAGE
144     !     REAL(SP) :: tbegin,tend  ! moved to mod_global 07/29/2016
145     
146     # if defined (PARALLEL)
147          CALL MPI_INIT ( ier )
148     # endif
149     
150          CALL READ_INPUT
151     
152     	![ykchoi(14.12.24.)
153     	IF(INI_UVZ)THEN
154             TIME=HotStartTime
155           ENDIF
156     	!ykchoi(14.12.24.)]
157     
158          CALL INDEX
159     
160     ! allocate variables
161          CALL ALLOCATE_VARIABLES
162     
163          CALL INITIALIZATION
164           
165     
166     # if defined (COUPLING)
167          TIME=TIME_COUPLING_1
168     # endif
169     
170     # if defined (VESSEL)
171          CALL VESSEL_INITIAL
172     # endif
173     
174     ! time integration
175     
176          ! record wall time
177          CALL WALL_TIME_SECS(tbegin)
178     
179     
180        DO WHILE (TIME<TOTAL_TIME)
181     
182     
183     # if defined (SAMPLES)
184          IF(WaveMaker(1:7)=='LEF_SOL')THEN
185            CALL SOLITARY_WAVE_LEFT_BOUNDARY
186          ENDIF  
187     # endif 
188     
189     ! update three variables
190          Eta0=Eta
191          Ubar0=Ubar
192          Vbar0=Vbar  
193        
194          CALL UPDATE_MASK
195     
196          CALL EXCHANGE
197     
198     # if defined (COUPLING)
199              CALL OneWayCoupling
200     # endif
201     
202      
203          CALL ESTIMATE_DT(Mloc,Nloc,DX,DY,U,V,H,MinDepthFrc,DT,CFL,TIME)
204       
205     ! U0, V0 are moved to following part due to computation of Ut, Vt.
206     	U0=U   !ykchoi(15. 08. 06.)
207     	V0=V   !ykchoi
208     
209     # if defined (VESSEL)
210            CALL VESSEL_FORCING
211     # endif
212     
213          ! 3-ORDER RUNGE-KUTTA TIME STEPPING
214          DO ISTAGE=1,3
215     
216            IF(DISPERSION)THEN
217              CALL Cal_Dispersion
218            ENDIF 
219     
220            CALL FLUXES
221     
222     # if defined (WIND)
223             IF(WindForce)THEN
224               CALL WIND_EFFECT
225             ENDIF
226     # endif
227     
228            CALL SourceTerms   ! put sourceterms after fluxes in order to get eta_t
229     
230            CALL ESTIMATE_HUV(ISTAGE)     
231     
232     
233            CALL WAVE_BREAKING
234            
235            CALL EXCHANGE
236            
237     # if defined (COUPLING)
238            CALL OneWayCoupling
239     # endif
240              
241     
242            IF(WaveMaker(1:3)=='ABS') THEN
243              CALL ABSORBING_GENERATING_BC
244            ENDIF
245     
246            IF(DIRECT_SPONGE)THEN
247                CALL SPONGE_DAMPING
248            ENDIF
249     
250          ENDDO
251     
252          CALL MIXING_STUFF
253     
254     !  find maximum eta velocity 
255     
256           IF (OUT_Hmax.OR.OUT_Hmin.OR.OUT_Umax.OR.OUT_MFmax.OR.OUT_VORmax)THEN
257             CALL MAX_MIN_PROPERTY
258           ENDIF
259              
260     
261           CALL OUTPUT
262     
263           CALL CHECK_BLOWUP
264       
265        END DO
266     
267     # if defined (COUPLING)
268          CLOSE(11)
269     # endif
270     
271     # if defined (VESSEL)
272     
273     DO Kves = 201,200+NumVessel
274         CLOSE(Kves)
275     ENDDO
276     # endif
277     
278          ! record wall time at the end
279          call WALL_TIME_SECS(tend)
280     
281     # if defined (PARALLEL)
282          if(myid.eq.0) write(*,*) 'Simulation takes',tend-tbegin,'seconds'
283          if(myid.eq.0) write(3,*) 'Simulation takes',tend-tbegin,'seconds'
284          if (myid.eq.0) WRITE(*,*)'Normal Termination!'
285          if (myid.eq.0) WRITE(3,*)'Normal Termination!'
286     # else
287          write(*,*) 'Simulation takes',tend-tbegin,'seconds'
288          write(3,*) 'Simulation takes',tend-tbegin,'seconds'
289          WRITE(*,*)'Normal Termination!'
290          WRITE(3,*)'Normal Termination!'
291     # endif
292     
293     # if defined (PARALLEL)
294          call MPI_FINALIZE ( ier )
295     # endif
296     
297     END PROGRAM FUNWAVE_TVD

306     !
307     ! Depth(): still water depth at element point
308     ! DepthNode(): still water depth at node
309     ! DepthX(): still water depth at x-interface
310     ! DepthY(): still water depth at y-interface
311     ! Eta():   surface elevation
312     ! Eta0(): Eta at previous time level
313     !  for dry point, Eta() = MinDepth+Z()
314     ! MASK(): 1 - wet
315     !         0 - dry
316     ! MASK_STRUC(): 0 - permanent dry point
317     ! MASK9: mask for itself and 8 elements around
318     ! 
319     ! U():  depth-averaged u or u at the reference level (u_alpha) at element
320     ! V():  depth-averaged v or v at the reference level (v_alpha) at element
321     ! HU(): (dep+eta)*u at element
322     ! HV(): (dep+eta)*v at element
323     ! P(): HU + dispersion at x-interface
324     ! Q(): HV + dispersion at y-interface
325     ! Fx(): F at x-interface
326     ! Fy(): F at y-interface
327     ! Gx(): G at x-interface
328     ! Gy(): G at y-interface
329     ! Ubar(:,:,:): Ubar
330     ! Vbar(:,:,:): Vbar
331     
332     ! dispersion
333     ! U1p(:,:): x-component of V1p
334     ! V1p(:,:): y-component of V1p
335     
336     ! 
337     ! EtaRxL(): Eta Left value at x-interface
338     ! EtaRxR(): Eta Right value at x-interface
339     ! EtaRyL(): Eta Left value at y-interface
340     ! EtaRyR(): Eta Right value at y-interface
341     ! HxL():   total depth  Left value at x-interface
342     ! HxR():   total depth  Right value at x-interface
343     ! HyL():   total depth  Left value at y-interface
344     ! HyR():   total depth  Right value at y-interface
345     
346     ! HUxL(): HU Left value at x-interface
347     ! HUxR(): HU Right value at x-interface
348     ! HUyL(): HV Left value at y-interface
349     ! HUyR(): HV Right value at y-interface
350     
351     ! PL(): HU + dispersion, Left value at x-interface
352     ! PR(): HU + dispersion, Right value at x-interface
353     ! QL(): HV + dispersion, Left value at y-interface
354     ! QR(): HV + dispersion, Right value at y-interface
355     
356     ! FxL = HUxL*UxL + 1/2*g*(EtaRxL^2 + 2*EtaRxL*Depthx)
357     ! FxR = HUxR*UxR + 1/2*g*(EtaRxR^2 + 2*EtaRxR*Depthx)
358     ! FyL = HyL*UyL*VyL
359     ! FyR = HyR*UyR*VyR
360     
361     ! GxL = HxL*UxL*VxL
362     ! GxR = HxR*UxR*VxR
363     ! GyL = HVyL*VyL + 1/2*g*(EtaRyL^2 + 2*EtaRyL*Depthy)
364     ! GyR = HVyR*VyR + 1/2*g*(EtaRyR^2 + 2*EtaRyR*Depthy)
