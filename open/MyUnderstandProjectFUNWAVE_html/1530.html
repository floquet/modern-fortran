54     SUBROUTINE INDEX
55         USE GLOBAL
56         IMPLICIT NONE
57         INTEGER :: icp
58     
59     ! TEMP
60     # if defined (PARALLEL)
61     
62         ALLOCATE(ProcessorID(PX,PY)) 
63     
64         NumberProcessor = px*py
65         dims(1) = px
66         dims(2) = py
67         periods(1) = .false.
68         periods(2) = .false.
69     
70     ! dont use periodic because our TRID solver is not based on
71     !  the cyclic topology  
72     !    IF(PERIODIC) periods(2) = .true.  
73     
74         coords(1) = 0
75         coords(2) = 0
76     
77         call MPI_CART_CREATE( MPI_COMM_WORLD, ndims, dims, &
78              periods, reorder, comm2d, ier )
79         call MPI_CART_COORDS( comm2d, myid, 2, coords, ier)
80     
81         npx = coords(1)
82         npy = coords(2)
83     
84         call MPI_Cart_shift( comm2d, 0, 1, n_west, n_east, ier )
85         call MPI_Cart_shift( comm2d, 1, 1, n_suth, n_nrth, ier )
86     
87         icp=0
88         DO I=1,PX
89         DO J=1,PY
90           ProcessorID(I,J) = icp
91           icp=icp+1
92         ENDDO
93         ENDDO
94     
95     ! check
96     ! print*,myid, n_west,n_east,n_suth,n_nrth,ProcessorID(1,1),ProcessorID(1,2)
97     !      print*,myid,ProcessorID(1,1),ProcessorID(2,1),ProcessorID(1,2),ProcessorID(2,2)
98     !      call MPI_FINALIZE ( ier )
99     
100     # else
101       px=1
102       py=1
103     # endif
104     
105     ! now for serial code
106         Mloc=Mglob/px+2*Nghost
107         Nloc=Nglob/py+2*Nghost
108         Mloc1=Mloc+1
109         Nloc1=Nloc+1
110     
111         Ibeg=Nghost+1
112         Iend=Mloc-Nghost
113         Iend1=Mloc1-Nghost
114         Jbeg=Nghost+1
115         Jend=Nloc-Nghost
116         Jend1=Nloc1-Nghost
117     
118     END SUBROUTINE INDEX

121     !
122     !    ESTIMATE_DT is subroutine evaluate dt based in CFL
123     !
124     !    HISTORY: 
125     !       05/06/2010 Fengyan Shi
126     !
127     !-------------------------------------------------------------------------------------
128     
129     SUBROUTINE ESTIMATE_DT(M,N,DX,DY,U,V,H,MinDepthFrc,DT,CFL,TIME)
130          USE PARAM
131     # if defined (PARALLEL)
132          USE GLOBAL, ONLY : ier
133     # endif
134          IMPLICIT NONE
135          INTEGER,INTENT(IN)::M,N
136     # if defined (PARALLEL)
137          REAL(SP) :: myvar
138     # endif
139     
140     # if defined (CARTESIAN)
141          REAL(SP),INTENT(IN)::DX,DY
142     # else
143          REAL(SP),DIMENSION(M,N),INTENT(IN)::DX,DY
144     # endif
145          REAL(SP),INTENT(IN),DIMENSION(M,N)::U,V,H
146          REAL(SP),INTENT(IN)::CFL,MinDepthFrc
147          REAL(SP),INTENT(OUT)::DT
148          REAL(SP),INTENT(INOUT)::TIME
149     
150          TMP3=LARGE
151          DO J=1,N
152          DO I=1,M
153     ! x direction
154           TMP1=ABS(U(I,J))+SQRT(GRAV*MAX(H(I,J),MinDepthFrc))
155           IF(TMP1<SMALL)THEN
156     # if defined (CARTESIAN)
157            TMP2=DX/SMALL
158     # else
159            TMP2=DX(I,J)/SMALL
160     # endif
161           ELSE
162     # if defined (CARTESIAN)
163            TMP2=DX/TMP1
164     # else
165            TMP2=DX(I,J)/TMP1
166     # endif
167           ENDIF
168           IF(TMP2<TMP3)TMP3=TMP2
169     ! y direction
170           TMP1=ABS(V(I,J))+SQRT(GRAV*MAX(H(I,J),MinDepthFrc))
171           IF(TMP1<SMALL)THEN
172     # if defined (CARTESIAN)
173            TMP2=DY/SMALL
174     # else
175            TMP2=DY(I,J)/SMALL
176     # endif
177           ELSE
178     # if defined (CARTESIAN)
179            TMP2=DY/TMP1
180     # else
181            TMP2=DY(I,J)/TMP1
182     # endif
183           ENDIF
184           IF(TMP2<TMP3)TMP3=TMP2      
185          ENDDO
186          ENDDO
187     # if defined (PARALLEL)
188          call MPI_ALLREDUCE (TMP3,myvar,1,MPI_SP,MPI_MIN,&
189               MPI_COMM_WORLD,ier)
190          TMP3 = myvar
191     # endif
192          DT=CFL*TMP3
193     ! TEMP
194          TIME=TIME+DT
195     
196     END SUBROUTINE ESTIMATE_DT

199     !-------------------------------------------------------------------------------------
200     !
201     !    WIND_EFFECT is subroutine to update wind and mask_wind 
202     !        based on Chen et al., 2004
203     !
204     !    HISTORY: 
205     !        09/16/2011 Fengyan Shi
206     !
207     !-------------------------------------------------------------------------------------
208     SUBROUTINE WIND_EFFECT
209         USE GLOBAL
210         IMPLICIT NONE
211          REAL(SP)::Celerity,WaveAngle
212     
213     ! time series wind
214           IF(TIME>TimeWind(icount_winddata).AND. &
215                    icount_winddata<NumTimeWindData)THEN
216             icount_winddata=icount_winddata+1
217           ENDIF
218           IF(icount_winddata>1)THEN  ! wind start
219            IF(TIME>TimeWind(icount_winddata))THEN
220             tmp2=ZERO
221            ELSE
222             tmp2=(TimeWind(icount_winddata)-TIME) &
223                /(TimeWind(icount_winddata)-TimeWind(icount_winddata-1))
224            ENDIF
225            WindU2D=WU(icount_winddata)*(1.0_SP-tmp2)&
226                            +WU(icount_winddata-1)*tmp2
227            WindV2D=WV(icount_winddata)*(1.0_SP-tmp2)&
228                            +WV(icount_winddata-1)*tmp2
229           ENDIF
230     
231     ! adjusted by wave celerity based on Chen et al. 2004
232     
233          DO J=1,Nloc
234          DO I=1,Mloc
235     
236          tmp3=SQRT(GRAV*MAX(MinDepthFrc,H(I,J)))
237          tmp1=MAX(SQRT(ETAx(I,J)*ETAx(I,J)+ETAy(I,J)*ETAy(I,J)),SMALL)
238     
239          Celerity=MIN(ABS(ETAt(I,J))/tmp1,SQRT(GRAV*ABS(H(I,J))))
240          WaveAngle=ATAN2(ETAy(I,J),ETAx(I,J))
241     
242          WindU2D(I,J)=WindU2D(I,J)-Celerity*COS(WaveAngle)
243          WindV2D(I,J)=WindV2D(I,J)-Celerity*SIN(WaveAngle)
244     
245          ENDDO
246          ENDDO
247     
248     ! mask_wind
249          MASK_WIND=1
250          DO J=1,Nloc
251          DO I=1,Mloc
252           IF(ETA(I,J)<HeightMax(I,J)*(1-WindCrestPercent))MASK_WIND(I,J)=0
253          ENDDO
254          ENDDO
255     
256     END SUBROUTINE WIND_EFFECT

259     !-------------------------------------------------------------------------------------
260     !
261     !    MAX_MIN_PROPERTY is subroutine to calculate Max and Min properties 
262     !        based on Chen et al., 2004
263     !
264     !    HISTORY: 
265     !        02/10/2016 Fengyan Shi
266     !
267     !-------------------------------------------------------------------------------------
268     SUBROUTINE MAX_MIN_PROPERTY
269         USE GLOBAL
270         IMPLICIT NONE
271         REAL(SP) :: maxv, MaxAbsEta
272     
273           DO J=1,Nloc
274           DO I=1,Mloc
275            IF(OUT_Hmax)THEN
276             IF(MASK(I,J).GT.0)THEN
277             IF(Eta(I,J).GT.HeightMax(I,J)) HeightMax(I,J)=Eta(I,J)
278             ENDIF
279            ENDIF
280     
281            IF(OUT_Hmin)THEN
282             IF(MASK(I,J).GT.0)THEN
283             IF(Eta(I,J).LT.HeightMin(I,J)) HeightMin(I,J)=Eta(I,J)
284             ENDIF
285            ENDIF
286     
287            IF(OUT_Umax)THEN
288             IF(MASK(I,J).GT.0)THEN
289               maxv=SQRT(U(I,J)*U(I,J)+V(I,J)*V(I,J))
290               IF(maxV.GT.VelocityMax(I,J)) VelocityMax(I,J)=maxV
291             ENDIF
292            ENDIF
293     
294            IF(OUT_MFmax)THEN
295             IF(MASK(I,J).GT.0)THEN
296               maxv=(U(I,J)*U(I,J)+V(I,J)*V(I,J))*(H(I,J))
297               IF(maxv.GT.MomentumFluxMax(I,J)) MomentumFluxMax(I,J)=maxv
298             ENDIF
299            ENDIF
300     
301     
302           ENDDO
303           ENDDO
304     
305     # if defined (PARALLEL)
306           IF(OUT_VORmax) THEN
307            CALL phi_exch(VorticityMax)
308           ENDIF
309     # endif   
310     
311     
312     END SUBROUTINE MAX_MIN_PROPERTY

362     !-------------------------------------------------------------------------------------
363     !
364     !   wall_time_secs is used to calculate current wall time
365     !
366     !   HISTORY: 
367     !   Gangfeng Ma, 09/12/2011
368     !
369     !-------------------------------------------------------------------------------------
370         SUBROUTINE wall_time_secs(tcurrent)
371         IMPLICIT NONE
372         INTEGER, dimension(8) :: walltime
373         real, INTENT(OUT) :: tcurrent
374         real :: msecs,secs,mins,hrs,days,months,mscale,years
375     
376         call date_and_time(VALUES=walltime)
377     
378         msecs = real(walltime(8))
379         secs = real(walltime(7))
380         mins = real(walltime(6))
381         hrs = real(walltime(5))
382         days = real(walltime(3))
383         months = real(walltime(2))
384         years = real(walltime(1))
385     
386         if((months.eq.1).or.(months.eq.3).or.(months.eq.5).or.  &
387               (months.eq.7).or.(months.eq.8).or.(months.eq.10).or.  &                                                                                   
388               (months.eq.12)) then
389           mscale = 31.0
390         elseif((months.eq.4).or.(months.eq.6).or.  &
391               (months.eq.9).or.(months.eq.11)) then
392           mscale = 30.0
393         elseif(years.eq.4*int(years/4)) then
394           mscale = 29.0
395         else
396           mscale = 28.0
397         endif
398     
399         tcurrent = months*mscale*24.0*60.0*60.0+days*24.0*60.0*60.0+  &
400              hrs*60.0*60.0+60.0*mins+secs+msecs/1000.0
401     
402         return
403         end SUBROUTINE wall_time_secs

406     SUBROUTINE SHAPIRO_3(M,N,Nghost,MASK,VAR_IN_OUT)  
407       USE PARAM,ONLY : SP
408       IMPLICIT NONE
409       REAL(SP),PARAMETER :: a1=0.68750_SP    ! 44/64
410       REAL(SP),PARAMETER :: a2=0.234375_SP   ! 15/64
411       REAL(SP),PARAMETER :: a3=-0.09375_SP   ! -6/64
412       REAL(SP),PARAMETER :: a4=0.015625_SP        ! 1/64
413       INTEGER,          INTENT(IN) :: M,N,Nghost
414       REAL(SP),DIMENSION(M,N),INTENT(INOUT) :: VAR_IN_OUT
415       REAL(SP),DIMENSION(M,N) :: VAR
416       INTEGER, DIMENSION(M,N),INTENT(IN) :: MASK
417       REAL(SP) :: p0,pp1,pp2,pp3,pn1,pn2,pn3
418       INTEGER :: I,J
419     
420     
421     ! x direction
422       DO J=1,N
423       DO I=1+Nghost,M-Nghost
424         IF(MASK(I,J)==1) THEN
425           p0=VAR_IN_OUT(I,J)
426           pp1=VAR_IN_OUT(I+1,J)*MASK(I+1,J)+(1.0-MASK(I+1,J))*p0
427           pp2=VAR_IN_OUT(I+2,J)*MASK(I+2,J)+(1.0-MASK(I+2,J))*p0
428           pp3=VAR_IN_OUT(I+3,J)*MASK(I+3,J)+(1.0-MASK(I+3,J))*p0
429           pn1=VAR_IN_OUT(I-1,J)*MASK(I-1,J)+(1.0-MASK(I-1,J))*p0
430           pn2=VAR_IN_OUT(I-2,J)*MASK(I-2,J)+(1.0-MASK(I-2,J))*p0
431           pn3=VAR_IN_OUT(I-3,J)*MASK(I-3,J)+(1.0-MASK(I-3,J))*p0
432     
433           VAR(I,J) = a1*p0+a2*(pp1+pn1)+a3*(pp2+pn2)+a4*(pp3+pn3)
434     
435         ENDIF
436       ENDDO
437       ENDDO
438     
439     
440     ! y direction
441     
442       DO J=1+Nghost,N-Nghost
443       DO I=1,M
444         IF(MASK(I,J)==1) THEN
445           p0=VAR(I,J)
446           pp1=VAR(I,J+1)*MASK(I,J+1)+(1.0-MASK(I,J+1))*p0
447           pp2=VAR(I,J+2)*MASK(I,J+2)+(1.0-MASK(I,J+2))*p0
448           pp3=VAR(I,J+3)*MASK(I,J+3)+(1.0-MASK(I,J+3))*p0
449           pn1=VAR(I,J-1)*MASK(I,J-1)+(1.0-MASK(I,J-1))*p0
450           pn2=VAR(I,J-2)*MASK(I,J-2)+(1.0-MASK(I,J-2))*p0
451           pn3=VAR(I,J-3)*MASK(I,J-3)+(1.0-MASK(I,J-3))*p0
452     
453           VAR_IN_OUT(I,J) = a1*p0+a2*(pp1+pn1)+a3*(pp2+pn2)+a4*(pp3+pn3)
454     
455         ENDIF
456       ENDDO
457       ENDDO
458     
459     
460     END SUBROUTINE SHAPIRO_3
