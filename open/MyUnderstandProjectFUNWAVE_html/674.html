66

71                                            TimeVessel1,TimeVessel2,ThetaVessel

85     ! READ VESSEL
86     
87     SUBROUTINE VESSEL_INITIAL
88       USE GLOBAL,ONLY : itmp1,itmp2,itmp3,itmp4
89                         
90       USE INPUT_READ
91       IMPLICIT NONE
92       CHARACTER(LEN=80)::FILE_NAME=' '
93       CHARACTER(LEN=80)::FILE_VESSEL=' '
94       CHARACTER(LEN=80) :: VESSEL_FOLDER=' '
95       CHARACTER(LEN=80)::TMP_NAME=' '
96       INTEGER :: Ifile,ierr
97       CHARACTER(LEN=80):: VesselName
98     
99     ! read vessel number and folder from input.txt
100           FILE_NAME='input.txt'
101     
102     ! vessel folder
103           CALL READ_STRING(VESSEL_FOLDER,FILE_NAME,'VESSEL_FOLDER',ierr)
104     # if defined (PARALLEL)
105           if (myid.eq.0) WRITE(3,'(A15,A50)')'VESSEL_FOLDER:', VESSEL_FOLDER
106     # else
107           WRITE(3,'(A15,A50)')'VESSEL_FOLDER:', VESSEL_FOLDER
108     # endif
109     
110           CALL READ_INTEGER(NumVessel,FILE_NAME,'NumVessel',ierr)
111     # if defined (PARALLEL)
112           if (myid.eq.0) WRITE(3,'(A12,I3)') 'NumVessel = ',NumVessel
113     # else
114           WRITE(3,'(A12,I3)') 'NumVessel = ',NumVessel
115     # endif
116     
117           CALL READ_LOGICAL(OUT_VESSEL,FILE_NAME,'OUT_VESSEL',ierr)
118     
119           ALLOCATE (Xvessel1(NumVessel),Yvessel1(NumVessel),  &
120                     Xvessel2(NumVessel),Yvessel2(NumVessel),  &
121                     TimeVessel1(NumVessel),TimeVessel2(NumVessel), &
122                     LengthVessel(NumVessel),WidthVessel(NumVessel), &
123                     AlphaVessel(NumVessel),BetaVessel(NumVessel),  &
124                     Pvessel(NumVessel),ThetaVessel(NumVessel) )
125           ALLOCATE (VesselPressureTotal(Mloc,Nloc), VesselPressureEach(Mloc,Nloc),&
126                      Xco(Mloc),Yco(Nloc), VesselPressureX(Mloc,Nloc), &
127                      VesselPressureY(Mloc,Nloc) )
128     
129     
130     ! Xco, and Yco
131     
132     # if defined (PARALLEL)
133          Xco(Ibeg) = npx*(Mloc-2*Nghost)*DX
134     # else
135          Xco(Ibeg) = ZERO
136     # endif
137          DO I = Ibeg+1,Mloc
138            Xco(I) = Xco(I-1)+DX
139          ENDDO
140          DO I = Ibeg-1,Ibeg-Nghost,-1
141            Xco(I) = Xco(I+1)-DX
142          ENDDO
143     
144     # if defined (PARALLEL)
145          Yco(Jbeg) = npy*(Nloc-2*Nghost)*DY
146     # else
147          Yco(Jbeg) = ZERO
148     # endif
149          DO J = Jbeg+1,Nloc
150            Yco(J) = Yco(J-1)+DY
151          ENDDO
152          DO J = Jbeg-1,Jbeg-Nghost,-1
153            Yco(J) = Yco(J+1)-DY
154          ENDDO
155     
156       DO Kves = 1, NumVessel
157     
158     !  file name
159         itmp1=mod(Kves/1000,10)
160         itmp2=mod(Kves/100,10)
161         itmp3=mod(Kves/10,10)
162         itmp4=mod(Kves,10)
163         write(FILE_VESSEL(1:1),'(I1)')itmp1
164         write(FILE_VESSEL(2:2),'(I1)')itmp2
165         write(FILE_VESSEL(3:3),'(I1)')itmp3
166         write(FILE_VESSEL(4:4),'(I1)')itmp4
167     
168         TMP_NAME = TRIM(VESSEL_FOLDER)//'vessel_'//TRIM(FILE_VESSEL)
169     
170     ! check existing
171     
172      INQUIRE(FILE=TRIM(TMP_NAME),EXIST=FILE_EXIST)
173       IF(.NOT.FILE_EXIST)THEN
174     # if defined (PARALLEL)
175        IF(MYID==0)  &
176        WRITE(*,*) TRIM(TMP_NAME), ' specified in ', TRIM(VESSEL_FOLDER), ' but CANNOT BE FOUND. STOP'
177        CALL MPI_FINALIZE (ier)
178        STOP
179     # else
180         WRITE(*,*) TRIM(TMP_NAME), ' specified in ', TRIM(VESSEL_FOLDER), ' but CANNOT BE FOUND. STOP'
181         STOP
182     # endif
183       ENDIF
184     
185     ! open file
186       Ifile=Kves+200
187       OPEN(Ifile,FILE=TRIM(TMP_NAME))
188     
189     ! read file
190              READ(Ifile,*)  ! title
191              READ(Ifile,*)  VesselName  ! vessel name
192              READ(Ifile,*)  ! length and width
193              READ(Ifile,*)  LengthVessel(Kves), WidthVessel(Kves), &
194                           AlphaVessel(Kves),BetaVessel(Kves),Pvessel(Kves)
195              READ(Ifile,*)  ! t, x, y
196              READ(Ifile,*)  TimeVessel2(Kves),Xvessel2(Kves),Yvessel2(Kves)
197     
198              TimeVessel1(Kves) = TimeVessel2(Kves)
199              Xvessel1(Kves) = Xvessel2(Kves)
200              Yvessel1(Kves) = Yvessel2(Kves)
201     
202              AlphaVessel(Kves) = Max(SMALL, AlphaVessel(Kves))
203              BetaVessel(Kves) = Max(SMALL, BetaVessel(Kves))
204              AlphaVessel(Kves) = Min(1.0_SP, AlphaVessel(Kves))
205              BetaVessel(Kves) = Min(1.0_SP, BetaVessel(Kves))
206     
207     # if defined (PARALLEL)
208        IF(MYID==0)THEN
209        WRITE(3,*) 'Vessel Name: ',  TRIM(VesselName)
210        WRITE(3,*) 'Vessel Length', LengthVessel(Kves)
211        WRITE(3,*) 'Vessel Width', WidthVessel(Kves)
212        WRITE(3,*) 'Vessel Alpha', AlphaVessel(Kves)
213        WRITE(3,*) 'Vessel Beta', BetaVessel(Kves)
214        WRITE(3,*) 'Vessel P', PVessel(Kves)
215        WRITE(3,*) 'Initial Time, X, Y', TimeVessel2(Kves),Xvessel2(Kves),Yvessel2(Kves)
216        ENDIF
217     # else
218        WRITE(3,*) 'Vessel Name: ',  TRIM(VesselName)
219        WRITE(3,*) 'Vessel Length', LengthVessel(Kves)
220        WRITE(3,*) 'Vessel Width', WidthVessel(Kves)
221        WRITE(3,*) 'Vessel Alpha', AlphaVessel(Kves)
222        WRITE(3,*) 'Vessel Beta', BetaVessel(Kves)
223        WRITE(3,*) 'Vessel P', PVessel(Kves)
224        WRITE(3,*) 'Initial Time, X, Y', TimeVessel2(Kves),Xvessel2(Kves),Yvessel2(Kves)
225     # endif
226     
227       ENDDO  ! end Kves
228     
229     End SUBROUTINE VESSEL_INITIAL
230     
231     SUBROUTINE VESSEL_FORCING
232       USE GLOBAL,ONLY : Mloc,Nloc,tmp1,tmp2,SMALL,TIME,ZERO
233       USE INPUT_READ
234       IMPLICIT NONE
235       INTEGER :: Ifile,ierr,I,J
236       REAL(SP) :: Xves,Yves,Lves,Wves
237     
238       VesselPressureTotal = ZERO
239     
240       DO Kves = 1,NumVessel
241     
242         IF(TIME>TimeVessel1(Kves).AND.TIME>TimeVessel2(Kves)) THEN
243     
244              TimeVessel1(Kves)=TimeVessel2(Kves)
245              Xvessel1(Kves) = Xvessel2(Kves)
246              Yvessel1(Kves) = Yvessel2(Kves)
247     
248         Ifile = 200 + Kves
249     
250         READ(Ifile,*,END=120)  TimeVessel2(Kves),Xvessel2(Kves),Yvessel2(Kves)
251     
252         ThetaVessel(Kves) = ATAN2(Yvessel2(Kves)-Yvessel1(Kves),  &
253                                   Xvessel2(Kves)-Xvessel1(Kves))
254     
255     # if defined (PARALLEL)
256        IF(MYID==0)THEN
257          WRITE(3,*)'Read Vessel # ', Kves
258          WRITE(3,*)'T,X,Y = ', TimeVessel2(Kves),Xvessel2(Kves),Yvessel2(Kves)
259        ENDIF
260     # else
261          WRITE(3,*)'Read Vessel # ', Kves
262          WRITE(3,*)'T,X,Y = ', TimeVessel2(Kves),Xvessel2(Kves),Yvessel2(Kves)
263     # endif
264     
265         ENDIF ! end time > timevessel2
266     
267     ! calculate force
268         tmp2=ZERO
269         tmp1=ZERO
270     
271         IF(TIME>TimeVessel1(Kves))THEN
272           IF(TimeVessel1(Kves).EQ.TimeVessel2(Kves))THEN
273             ! no more data
274             tmp2=ZERO
275             tmp1=ZERO
276           ELSE
277           tmp2=(TimeVessel2(Kves)-TIME) &
278                 /MAX(SMALL, ABS(TimeVessel2(Kves)-TimeVessel1(Kves)))
279           tmp1=1.0_SP - tmp2;
280           ENDIF  ! no more data?
281         ENDIF ! time>time_1
282     
283         Xves = Xvessel2(Kves)*tmp1 +Xvessel1(Kves)*tmp2
284         Yves = Yvessel2(Kves)*tmp1 +Yvessel1(Kves)*tmp2
285     
286     ! rectangular
287         VesselPressureEach = ZERO
288         DO J=1,Nloc
289         DO I=1,Mloc
290           Lves=(Xco(I)-Xves)*COS(ThetaVessel(Kves)) + (Yco(J)-Yves)*SIN(ThetaVessel(Kves))
291           Wves=-(Xco(I)-Xves)*SIN(ThetaVessel(Kves)) + (Yco(J)-Yves)*COS(ThetaVessel(Kves))
292           IF(ABS(Lves)<=0.5_SP*LengthVessel(Kves).AND. &
293              ABS(Wves)<=0.5_SP*WidthVessel(Kves)) THEN
294              VesselPressureEach(I,J) = Pvessel(Kves)  &
295                       *COS(PI*Lves/(AlphaVessel(Kves)*LengthVessel(Kves)))**2 &
296                       *COS(PI*Wves/(BetaVessel(Kves)*WidthVessel(Kves)))**2                  
297           ENDIF
298         ENDDO
299         ENDDO
300     
301     120 CONTINUE  ! no more data for vessel Kves
302     
303         VesselPressureTotal = VesselPressureTotal+VesselPressureEach
304     
305       ENDDO  ! end Kves
306     
307     ! sourceX and sourceY
308     
309         DO J=Jbeg,Jend
310         DO I=Ibeg,Iend
311            VesselPressureX(I,J) = Grav*H(I,J)*  &
312                    (VesselPressureTotal(I+1,J)-VesselPressureTotal(I-1,J))/2.0_SP  &
313                    /DX
314            VesselPressureY(I,J) = Grav*H(I,J)*  &
315                    (VesselPressureTotal(I,J+1)-VesselPressureTotal(I,J-1))/2.0_SP  &
316                    /DY
317     
318         ENDDO
319         ENDDO
320     
321     
322     END SUBROUTINE VESSEL_FORCING
323     
324     END MODULE VESSEL_MODULE
325     
326     # endif 
327     ! end vessel
