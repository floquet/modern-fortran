100     SUBROUTINE FLUX_AT_INTERFACE
101          USE GLOBAL
102     
103     ! for averaging approach for predictor
104          P=0.5_SP*(PR+PL)
105          Fx=0.5_SP*(FxR+FxL)
106          Gx=0.5_SP*(GxR+GxL)
107          Q=0.5_SP*(QR+QL)
108          Fy=0.5_SP*(FyR+FyL)
109          Gy=0.5_SP*(GyR+GyL)

135     !-------------------------------------------------------------------------------------
136     !
137     !    HLL is subroutine for HLL scheme
138     !
139     !    HISTORY: 
140     !      05/06/2010 Fengyan Shi
141     !
142     !-------------------------------------------------------------------------------------
143     SUBROUTINE HLL(M,N,SL,SR,FL,FR,UL,UR,FOUT)
144          USE PARAM
145          INTEGER,INTENT(IN)::M,N
146          REAL(SP),INTENT(IN),DIMENSION(M,N)::SL,SR,FL,FR,UL,UR
147          REAL(SP),INTENT(OUT),DIMENSION(M,N)::FOUT
148     
149           DO J=1,N
150           DO I=1,M     
151           IF(SL(I,J)>=ZERO) THEN
152             FOUT(I,J)=FL(I,J)
153           ELSEIF(SR(I,J)<=ZERO) THEN
154             FOUT(I,J)=FR(I,J)
155           ELSE
156             FOUT(I,J)=SR(I,J)*FL(I,J)-SL(I,J)*FR(I,J)+SL(I,J)*SR(I,J)*(UR(I,J)-UL(I,J))
157             IF((ABS(SR(I,J)-SL(I,J)))<SMALL)THEN
158              FOUT(I,J)=FOUT(I,J)/SMALL
159             ELSE
160              FOUT(I,J)=FOUT(I,J)/(SR(I,J)-SL(I,J))
161             ENDIF
162           ENDIF
163           ENDDO
164           ENDDO
165     
166     END SUBROUTINE HLL

203     !-------------------------------------------------------------------------------------
204     SUBROUTINE DelyFun(DY,M,N,DIN,DOUT)
205          USE PARAM
206          IMPLICIT NONE
207     # if defined (CARTESIAN)
208          REAL(SP),INTENT(IN)::DY
209     # else
210          REAL(SP),DIMENSION(M,N),INTENT(IN)::DY
211     # endif
212          INTEGER,INTENT(IN)::M,N
213          REAL(SP),INTENT(IN),DIMENSION(M,N)::DIN
214          REAL(SP),INTENT(OUT),DIMENSION(M,N)::DOUT
215     !     REAL(SP) :: VANLEER_LIMITER
216     
217     ! van Leer Limiter     
218          DO I=1,M
219          DO J=2,N-1
220     # if defined (CARTESIAN)
221           TMP1=(DIN(I,J+1)-DIN(I,J))/DY
222           TMP2=(DIN(I,J)-DIN(I,J-1))/DY
223     # else
224           TMP1=(DIN(I,J+1)-DIN(I,J))/DY(I,J)
225           TMP2=(DIN(I,J)-DIN(I,J-1))/DY(I,J-1)
226     # endif
227           IF((ABS(TMP1)+ABS(TMP2))<SMALL)THEN
228     !       !WRITE(*,*)'WARNING Ã‰ (a+b) in limiter x is too small','(I,J)',I,J
229             DOUT(I,J)=ZERO
230           ELSE
231     !        DOUT(I,J)=(TMP1*ABS(TMP2)+ABS(TMP1)*TMP2)/(ABS(TMP1)+ABS(TMP2))
232     !        DOUT(I,J)=VANLEER_LIMITER(TMP1,TMP2)
233              DOUT(I,J)=(TMP1*ABS(TMP2)+ABS(TMP1)*TMP2)/(ABS(TMP1)+ABS(TMP2))
234           ENDIF
235          ENDDO
236          ENDDO
237     
238          DO I=1,M
239     # if defined (CARTESIAN)
240           DOUT(I,1)=(DIN(I,2)-DIN(I,1))/DY
241           DOUT(I,N)=(DIN(I,N)-DIN(I,N-1))/DY
242     # else
243           DOUT(I,1)=(DIN(I,2)-DIN(I,1))/DY(I,1)
244           DOUT(I,N)=(DIN(I,N)-DIN(I,N-1))/DY(I,N)
245     # endif
246          ENDDO
247     
248     END SUBROUTINE DelyFun

259          USE PARAM
260          IMPLICIT NONE
261     # if defined (CARTESIAN)
262          REAL(SP),INTENT(IN)::DX
263     # else
264          REAL(SP),DIMENSION(M,N),INTENT(IN)::DX
265     # endif
266          INTEGER,INTENT(IN)::M,N
267          REAL(SP),INTENT(IN),DIMENSION(M,N)::DIN
268          REAL(SP),INTENT(OUT),DIMENSION(M,N)::DOUT
269     !     REAL(SP) :: VANLEER_LIMITER
270     
271     ! van Leer Limiter     
272          DO I=2,M-1
273          DO J=1,N
274     # if defined (CARTESIAN)
275           TMP1=(DIN(I+1,J)-DIN(I,J))/DX
276           TMP2=(DIN(I,J)-DIN(I-1,J))/DX
277     # else
278           TMP1=(DIN(I+1,J)-DIN(I,J))/DX(I,J)
279           TMP2=(DIN(I,J)-DIN(I-1,J))/DX(I-1,J)
280     # endif
281           IF((ABS(TMP1)+ABS(TMP2))<SMALL)THEN
282     !       !WRITE(*,*)'WARNING Ã‰ (a+b) in limiter y is too small','(I,J)',I,J
283             DOUT(I,J)=ZERO
284           ELSE
285     !        DOUT(I,J)=VANLEER_LIMITER(TMP1,TMP2)
286              DOUT(I,J)=(TMP1*ABS(TMP2)+ABS(TMP1)*TMP2)/(ABS(TMP1)+ABS(TMP2))
287           ENDIF
288          ENDDO
289          ENDDO
290     
291          DO J=1,N
292     # if defined (CARTESIAN)
293           DOUT(1,J)=(DIN(2,J)-DIN(1,J))/DX
294           DOUT(M,J)=(DIN(M,J)-DIN(M-1,J))/DX
295     # else
296           DOUT(1,J)=(DIN(2,J)-DIN(1,J))/DX(1,J)
297           DOUT(M,J)=(DIN(M,J)-DIN(M-1,J))/DX(M,J)
298     # endif
299          ENDDO
300     
301     END SUBROUTINE DelxFun

303     ! ---------------------------------------------------
304     !
305     !    WAVE_SPEED is subroutine to calculate wave speed
306     !    no shear wave is calculated yet
307     !
308     !    HISTORY: 
309     !        01/21/2012 Fengyan Shi
310     !        10/29/2012 Fengyan Shi
311     !             Steve Brandt mentioned HxL and HxR not calculated 
312     !             inside ghost cells, it is not an issue though corrected
313     !             There's a bug found in Dmitry's case. In x direction, 
314     !             should use M1 and N1 for y direction.
315     !
316     ! --------------------------------------------------
317     SUBROUTINE WAVE_SPEED(M,N,M1,N1,UL,UR,VL,VR,HxL,HxR,HyL,HyR,&
318           SxL,SxR,SyL,SyR)
319          USE PARAM
320          USE GLOBAL, ONLY : Nghost
321          IMPLICIT NONE
322          INTEGER,INTENT(IN)::M,N,M1,N1
323          REAL(SP),INTENT(IN),DIMENSION(M1,N)::UL,UR,HxL,HxR
324          REAL(SP),INTENT(IN),DIMENSION(M,N1)::VL,VR,HyL,HyR
325          REAL(SP),INTENT(OUT),DIMENSION(M1,N)::SxL,SxR
326          REAL(SP),INTENT(OUT),DIMENSION(M,N1)::SyL,SyR         
327          REAL(SP)::SQR_PHI_L,SQR_PHI_R,SQR_PHI_S,U_S
328     
329     
330     ! Zhou et al., 2001 approach
331     ! x interface
332     !     DO J=1,N
333     !     DO I=1,M
334          DO J=1+Nghost,N-Nghost
335          DO I=1+Nghost,M1-Nghost
336            SQR_PHI_L=SQRT(GRAV*ABS(HxL(I,J)))
337            SQR_PHI_R=SQRT(GRAV*ABS(HxR(I,J)))
338            SQR_PHI_S=0.5*(SQR_PHI_L+SQR_PHI_R)+0.25*(UL(I,J)-UR(I,J))  
339            U_S=0.5*(UL(I,J)+UR(I,J))+SQR_PHI_L-SQR_PHI_R
340            SxL(I,J)=MIN(UL(I,J)-SQR_PHI_L,U_S-SQR_PHI_S)
341            SxR(I,J)=MAX(UR(I,J)+SQR_PHI_R,U_S+SQR_PHI_S)
342          ENDDO
343          ENDDO
344     
345     ! ghost cells, this does not really matter
346          DO J=1+Nghost,N-Nghost
347           DO I=1,Nghost
348            SxL(I,J)=SxL(Nghost+1,J)
349            SxR(I,J)=SxR(Nghost+1,J)
350           ENDDO
351           DO I=M1-Nghost+1,M1
352            SxL(I,J)=SxL(M1-Nghost,J)
353            SxR(I,J)=SxR(M1-Nghost,J)       
354           ENDDO
355          ENDDO
356     
357          DO I=1,M1
358            DO J=1,Nghost
359              SxL(I,J)=SxL(I,Nghost+1)
360              SxR(I,J)=SxR(I,Nghost+1)
361            ENDDO
362            DO J=N-Nghost+1,N
363              SxL(I,J)=SxL(I,N-Nghost)
364              SxR(I,J)=SxR(I,N-Nghost)
365            ENDDO
366          ENDDO
367     
368     ! y interface
369          DO J=1+Nghost,N1-Nghost
370          DO I=1+Nghost,M-Nghost
371            SQR_PHI_L=SQRT(GRAV*ABS(HyL(I,J)))
372            SQR_PHI_R=SQRT(GRAV*ABS(HyR(I,J)))
373            SQR_PHI_S=0.5*(SQR_PHI_L+SQR_PHI_R)+0.25*(VL(I,J)-VR(I,J))  
374            U_S=0.5*(VL(I,J)+VR(I,J))+SQR_PHI_L-SQR_PHI_R
375            SyL(I,J)=MIN(VL(I,J)-SQR_PHI_L,U_S-SQR_PHI_S)
376            SyR(I,J)=MAX(VR(I,J)+SQR_PHI_R,U_S+SQR_PHI_S)
377          ENDDO
378          ENDDO
379     
380     
381     ! ghost cells
382          DO I=1+Nghost,M-Nghost
383           DO J=1,Nghost
384            SyL(I,J)=SyL(I,Nghost+1)
385            SyR(I,J)=SyR(I,Nghost+1)
386           ENDDO
387           DO J=N1-Nghost+1,N1
388            SyL(I,J)=SyL(I,N1-Nghost)
389            SyR(I,J)=SyR(I,N1-Nghost)       
390           ENDDO
391          ENDDO
392     
393          DO J=1,N1
394            DO I=1,Nghost
395              SyL(I,J)=SyL(Nghost+1,J)
396              SyR(I,J)=SyR(Nghost+1,J)
397            ENDDO
398            DO I=M-Nghost+1,M
399              SyL(I,J)=SyL(M-Nghost,J)
400              SyR(I,J)=SyR(M-Nghost,J)
401            ENDDO
402          ENDDO
403     
404     END SUBROUTINE WAVE_SPEED

1491                TXP1=Vin(I-1,J)-Vin(I-2,J)

1507                else
